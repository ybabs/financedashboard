#!/usr/bin/env bash
# Conan-only build & GoogleTest runner for companies_house_ingestor
# - Requires Conan (v2) and uses Conan-generated CMake presets exclusively.
#

set -euo pipefail

# ---- config (Conan-only) ----
BUILD_TYPE="Debug"                  # or Release
BUILD_DIR="cmake-build-debug"
PRESET_NAME="conan-debug"
TARGET=""                           # e.g. ingestor_tests
RUNNER="gtest"                      # gtest | ctest
NO_CLEAN=0

# GTest flags (only used when RUNNER=gtest)
FILTER="*"
REPEAT="1"
BREAK=0
GTEST_OUTPUT=""                     # e.g. xml:./test-results/gtest.xml

# parallelism
if command -v nproc >/dev/null 2>&1; then
  NPROC=$(nproc)
elif command -v sysctl >/dev/null 2>&1; then
  NPROC=$(sysctl -n hw.ncpu)
else
  NPROC=4
fi
CORES=$(( NPROC > 4 ? NPROC - 4 : (NPROC > 0 ? NPROC : 1) ))

# ---- args ----
for arg in "$@"; do
  case "$arg" in
    --rel|--release) BUILD_TYPE="Release"; BUILD_DIR="cmake-build-release"; PRESET_NAME="conan-release";;
    --dbg|--debug)   BUILD_TYPE="Debug";   BUILD_DIR="cmake-build-debug";   PRESET_NAME="conan-debug";;
    --target=*)      TARGET="${arg#*=}";;
    --run=gtest)     RUNNER="gtest";;
    --no-clean)      NO_CLEAN=1;;
    --filter=*)      FILTER="${arg#*=}";;
    --repeat=*)      REPEAT="${arg#*=}";;
    --break)         BREAK=1;;
    --out=*)         GTEST_OUTPUT="${arg#*=}";;
    -j=*|--jobs=*)   CORES="${arg#*=}";;
    -h|--help)
      cat <<EOF
Conan-only build & test
  --debug|--dbg                 Debug build (default)
  --release|--rel               Release build
  --target=<name>               Build specific target (e.g. ingestor_tests)
  --no-clean                    Reuse build dir
  --run=gtest|ctest             Choose test runner (default: gtest)
  --filter=<pattern>            GTest filter (default: *)
  --repeat=<N>                  Repeat tests N times (default: 1)
  --break                       Break on first failure (gtest)
  --out=<spec>                  gtest output (e.g. xml:./test-results/gtest.xml)
  --jobs|-j=<N>                 parallel build jobs
EOF
      exit 0;;
    *) echo "Unknown option: $arg" >&2; exit 1;;
  esac
done

YELLOW='\033[1;33m'; GREEN='\033[0;32m'; RED='\033[0;31m'; NC='\033[0m'

# ---- guards: conan + preset must exist ----
if ! command -v conan >/dev/null 2>&1; then
  echo -e "${RED}Conan not found. Please install Conan (v2).${NC}" >&2
  exit 1
fi

# If a previous run hasn't generated the preset yet, that's fine; we'll run conan first.
# We still check after conan install below.

echo -e "${YELLOW}Conan-only pipeline | BuildType=${BUILD_TYPE} | Preset=${PRESET_NAME}${NC}"

# ---- 1. clean (optional) ----
if [[ $NO_CLEAN -eq 0 ]]; then
  echo -e "\n${GREEN}--> Cleaning '${BUILD_DIR}'...${NC}"
  rm -rf "${BUILD_DIR}"
else
  echo -e "\n${GREEN}--> Skipping clean (--no-clean).${NC}"
fi

# ---- 2. conan install (always) ----
echo -e "\n${GREEN}--> Conan install...${NC}"
conan install . -s build_type="${BUILD_TYPE}" --output-folder="${BUILD_DIR}" --build=missing

# ---- 3. cmake configure using the Conan-generated preset ----
echo -e "\n${GREEN}--> CMake configure with preset '${PRESET_NAME}'...${NC}"
cmake --preset "${PRESET_NAME}"

# ---- 4. build ----
echo -e "\n${GREEN}--> Building target '${TARGET:-all}' with ${CORES} cores...${NC}"
if [[ -z "${TARGET}" ]]; then
  cmake --build --preset "${PRESET_NAME}" --parallel "${CORES}"
else
  cmake --build --preset "${PRESET_NAME}" --target "${TARGET}" --parallel "${CORES}"
fi

# ---- 5. locate test dir + binary ----
# Conan preset typically builds into: ${BUILD_DIR}/build[/<CONFIG>]
BUILD_SUBDIR="${BUILD_DIR}/build"
CONFIG=${CMAKE_BUILD_TYPE:-${BUILD_TYPE}}

if [[ -d "${BUILD_SUBDIR}/${CONFIG}" ]]; then
  TEST_DIR="${BUILD_SUBDIR}/${CONFIG}"
elif [[ -d "${BUILD_SUBDIR}" ]]; then
  TEST_DIR="${BUILD_SUBDIR}"
else
  TEST_DIR="${BUILD_DIR}"
fi

TEST_BIN_CANDIDATES=(
  "${TEST_DIR}/unittests/ingestor_tests"
  "${TEST_DIR}/ingestor_tests"
)

TEST_BIN=""
for cand in "${TEST_BIN_CANDIDATES[@]}"; do
  if [[ -x "$cand" ]]; then TEST_BIN="$cand"; break; fi
done

# ---- 6. run tests (gtest or ctest), still Conan-built ----
echo -e "\n${GREEN}--> Running tests (${RUNNER})...${NC}"
if [[ "${RUNNER}" == "ctest" ]]; then
  ( cd "${TEST_DIR}" && ctest --output-on-failure -C "${BUILD_TYPE}" )
else
  if [[ -z "${TEST_BIN}" ]]; then
    echo -e "${RED}Couldn't find 'ingestor_tests' binary under ${TEST_DIR}.${NC}" >&2
    echo "Make sure unittests/CMakeLists.txt builds an 'ingestor_tests' target."
    exit 2
  fi
  export GTEST_COLOR=1
  CMD=( "${TEST_BIN}" "--gtest_filter=${FILTER}" "--gtest_repeat=${REPEAT}" )
  [[ $BREAK -eq 1 ]] && CMD+=( "--gtest_break_on_failure" )
  [[ -n "${GTEST_OUTPUT}" ]] && CMD+=( "--gtest_output=${GTEST_OUTPUT}" )
  echo "Running: ${CMD[*]}"
  "${CMD[@]}"
fi

echo -e "\n${YELLOW}Conan-only build & tests completed.${NC}"

#include "DeltaPoll.h"

#include "JsonParse.h"
#include "Pg.h"

namespace delta

    std::optional<std::string> getWatermark(pqxx::connection& conn, const std::string& source)
    {
        pqxx::work w(conn);
        auto r = w.exec("SELECT watermark FROM ingest_watermarks WHERE source=$1", pqxx::params{source});
        w.commit();
        if (r.empty())
        {
            return std::nullopt;
        }
        return r[0][0].c_str();
    }


    void setWatermark(pqxx::connection& conn, const std::string& source, const std::string& isoTs)
    {
        pqxx::work w(conn);
        w.exec("INSERT INTO ingest_watermarks(source, watermark) VALUES($1,$2)"
            "ON CONFLICT (source) DO UPDATE SET watermark=EXCLUDED.watermark",
            pqxx::params{source, isoTs}).no_rows();
        w.commit();
    }


    static void upsertOneCompany(pqxx::connection& conn, ChApi& ch, const std::string& companyNo, DeltaStats& stats)
    {
        try
        {
            auto compj = ch.getCompany(companyNo);
            CompanyRecord record;

            {
                const std::string s = compj.dump();
                const std::string err = parseCompanyRest(s, record);
                if (!err.empty())
                {
                    throw std::runtime_error("parse_company_rest" + err);
                }
            }

            auto officersJ = ch.getAllOfficers(companyNo);

            std::vector<OfficerRecord> officerRecords;
            {
                nlohmann::json root;
                root["items"] = officersJ;
                const std::string s = root.dump();
                std::string err = parseOfficersRest(s, companyNo, officerRecords);
                if (!err.empty())
                {
                    throw std::runtime_error("parse_officers_rest" + err);
                }
            }

            auto pscJsonVec = ch.getAllPSC(companyNo);
            std::vector<PscRecord> pscRecords;
            {
                nlohmann::json root;
                root["items"] = pscJsonVec;
                const std::string s = root.dump();
                std::string err = parsePSCRest(s, companyNo, pscRecords);
                if (!err.empty())
                {
                    throw std::runtime_error("parse_psc_rest" + err);
                }
            }

            pg::prepareStatements(conn);
            {
                pqxx::work w{conn};
                pg::upsertCompany(w, record.companyNumber, record.name, record.status, record.incorporationDate, record.sicCodes,
                    record.region, record.lastAccountsMadeUpTo, record.accountType);

                std::vector<
                std::tuple<std::string,std::string,
                           std::optional<std::string>,
                           std::optional<std::string>,
                           std::optional<int>>> officerRows;
                officerRows.reserve(officerRecords.size());

                for (auto& o : officerRecords)
                {
                    officerRows.emplace_back(o.name, o.role, o.appointedOn, o.resignedOn, o.dobYear);
                }
                pg::replaceOfficers(w, record.companyNumber, officerRows);
                stats.officersWritten += officerRows.size();

                pg::replacePSC(w, record.companyNumber, pscRecords);
                stats.pscWritten += pscRecords.size();

                w.commit();
            }

            stats.companiesOk++;
        }

        catch (const std::exception& e)
        {
            // TODO Log here
            stats.companiesFailed++;
        }
        stats.companiesSeen++;
    }
    bool runForCompanyList(pqxx::connection& conn, ChApi& ch, const std::vector<std::string>& companyNumbers, DeltaStats* out, int throttleMs)
    {
        pg::ensureMinimalSchema(conn);
        pg::prepareStatements(conn);

        DeltaStats stats{};
        for (const auto& num: companyNumbers)
        {
            upsertOneCompany(conn, ch, num, stats);
            if (throttleMs > 0)
            {
                std::this_thread::sleep_for(std::chrono::milliseconds(throttleMs));
            }
        }
        if (out)
        {
            *out = stats;
        }

        return true;

    }


    bool runSearchByPage(pqxx::connection& conn, ChApi& ch, const std::string& q, int maxPages, int itemsPerPage, DeltaStats* out, int throttleMs)
    {

        pg::ensureMinimalSchema(conn);
        pg::prepareStatements(conn);
        DeltaStats stats{};
        int start = 0;
        for (int page = 0; page < maxPages; ++page)
        {
            auto pageJ = ch.searchCompanies(q, start, itemsPerPage);
            auto numbers = ChApi::extractCompanyNumbersFromSearch(pageJ);
            if (numbers.empty())
            {
                break;
            }

            for (const auto& number : numbers)
            {
                upsertOneCompany(conn, ch, number, stats);
                if (throttleMs > 0)
                {
                    std::this_thread::sleep_for(std::chrono::milliseconds(throttleMs));
                }
            }

            if ((int)numbers.size() < itemsPerPage)
            {
                break;
            }
            start+=itemsPerPage;
        }
        if (out)
        {
            *out = stats;
        }
        return true;
    }

#include <thread>

#include "Http.h"

#include <random>

Http::Http(int timeoutSeconds, int maxRetries, std::chrono::milliseconds retryBackoff):
    mMaxRetries(maxRetries), mTimeoutSeconds(timeoutSeconds), mRetryBackoffBase(retryBackoff)
{

}

void Http::setDefaultHeaders(const cpr::Header& headers)
{
    mDefaultHeaders = headers;
}

void Http::setBasicAuth(const std::string& user, const std::string& password)
{
    mDefaultAuth = cpr::Authentication{user, password, cpr::AuthMode::BASIC};
}


static inline uint64_t randJitterMs(uint64_t baseMs, uint64_t jitterFracPercent = 20)
{
    static thread_local std::mt19937 rng{std::random_device{}()};
    const int64_t span = static_cast<int64_t>(baseMs * jitterFracPercent/100);
    std::uniform_int_distribution<int64_t>dist(-span , span);
    int64_t j = dist(rng);
    int64_t out = static_cast<int64_t>(baseMs) + j;
    return out < 0 ? 0 : static_cast<uint64_t>(out);
}

std::chrono::milliseconds Http::calculateBackoff(int attempt) const
{
    // attempt starts at 1
    uint64_t base = mRetryBackoffBase.count();
    // exponential backoff  base * 2 ^ (attempt - 1), clamp t 30s
    uint64_t ms = base << (attempt -1);
    ms = std::min<uint64_t>(ms, 30000);
    ms = randJitterMs(ms, 20);
    ms = std::min<uint64_t>(ms, 30000);
    return std::chrono::milliseconds{ms};
}

bool Http::shouldRetry(const cpr::Response& response) const
{
    if (response.error.code != cpr::ErrorCode::OK)
    {
        return true;
    }
    const long sc = response.status_code;
    if (sc == 0) return true;
    if (sc == 408) return true;
    if (sc == 429) return true;
    if (sc >= 500 && sc <= 599) return true;
    return false;
}

std::optional<int> Http::retryAfterSeconds(const cpr::Response& response)
{
    auto it = response.header.find("Retry-After");
    if (it == response.header.end())
    {
        return std::nullopt;
    }

    try
    {
         int s = std::stoi(it->second);
        if (s >= 0 && s <= 3600)
        {
            return s;
        }
    }
    catch (...)
    {
    }
    return std::nullopt;
}

void Http::sleepFor(std::chrono::milliseconds ms)
{
    std::this_thread::sleep_for(ms);
}

cpr::Header Http::mergeHeaders(const cpr::Header& a, const cpr::Header& b)
{
    cpr::Header ret = a;
    for (const auto& kv: b)
    {
        ret[kv.first] = kv.second;
    }

    return ret;
}

bool Http::getText(const std::string& url, const cpr::Header& headers, std::string& out) const
{
    out.clear();
    cpr::Header hdr = mergeHeaders(mDefaultHeaders, headers);

    for (int attempt = 1; attempt <= std::max<int>(mMaxRetries, 1); ++attempt)
    {
        cpr::Session session;
        session.SetUrl(url);
        session.SetHeader(hdr);
        session.SetTimeout(mTimeoutSeconds * 1000);
        if (mDefaultAuth)
        {
            session.SetAuth(*mDefaultAuth);
        }

        cpr::Response resp = session.Get();
        const bool success = (resp.error.code == cpr::ErrorCode::OK) && (resp.status_code >= 200 && resp.status_code < 300);
        if (success)
        {
            out = std::move(resp.text);
            return true;
        }

        if (attempt >= mMaxRetries || !shouldRetry(resp))
        {
            out.clear();
            return false;
        }
        if (auto ra = retryAfterSeconds(resp))
        {
            sleepFor(std::chrono::seconds(*ra));
        }
        else
        {
            sleepFor(calculateBackoff(attempt));
        }
    }

    out.clear();
    return false;
}

HttpResponse Http::get(const std::string& url, const cpr::Header& headers, const cpr::Parameters& params) const
{
    HttpResponse result;
    cpr::Header hdr = mergeHeaders(mDefaultHeaders, headers);

    for (int attempt = 1; attempt <= std::max(mMaxRetries, 1); ++attempt)
    {
        cpr::Session session;
        session.SetUrl(url);
        session.SetHeader(hdr);
        session.SetTimeout(mTimeoutSeconds * 1000);
        session.SetParameters(params);
        if (mDefaultAuth)
        {
            session.SetAuth(*mDefaultAuth);
        }
        cpr::Response resp = session.Get();
        result.statusCode = resp.status_code;
        result.body = std::move(resp.text);
        result.headers = resp.header;
        result.success = (resp.error.code == cpr::ErrorCode::OK) && (resp.status_code >= 200 && resp.status_code < 300 );
        result.errorMessage = (resp.error.code == cpr::ErrorCode::OK) ? " " : resp.error.message ;

        if (result.success)
        {
            return result;
        }

        if (attempt >= mMaxRetries || !shouldRetry(resp))
        {
            return result;
        }

        if (auto ra = retryAfterSeconds( resp))
        {
            sleepFor(std::chrono::seconds(*ra));
        }
        else
        {
            sleepFor(calculateBackoff(attempt));
        }
    }

    return result;
}

        result.errorMessage = (resp.error.code == cpr::ErrorCode::OK) ? " " : resp.error.message ;

bool Http::downloadFile(const std::string& url, const std::string& filePath, const cpr::Header& headers, std::function<void(size_t, size_t)> progressCallback)
{
    cpr::Header hdr = mergeHeaders(mDefaultHeaders, headers);

    for (int attempt = 1; attempt <= std::max(mMaxRetries, 1); ++attempt)
    {
        std::ofstream ofs(filePath, std::ios::binary|std::ios::trunc);
        if (!ofs)
        {
            return false;
        }
        cpr::Session session;
        session.SetUrl(url);
        session.SetHeader(hdr);
        session.SetTimeout(mTimeoutSeconds * 1000);
        if (mDefaultAuth)
        {
            session.SetAuth(*mDefaultAuth);
        }
        size_t totalBytes = 0;
        session.SetWriteCallback(cpr::WriteCallback{
            [&ofs, &totalBytes, &progressCallback](std::string data, intptr_t)
            {
                if (!data.empty())
                {
                    ofs.write(data.data(), static_cast<std::streamsize>(data.size()));
                    totalBytes += data.size();
                    if (progressCallback)
                    {
                        progressCallback(totalBytes, 0);
                    }
                }
                return static_cast<bool>(ofs);
            }
        });

        if (progressCallback)
        {
            session.SetProgressCallback(cpr::ProgressCallback{
                [&progressCallback](cpr::cpr_off_t dltotal, cpr::cpr_off_t dlnow,
                                    cpr::cpr_off_t, cpr::cpr_off_t, intptr_t) {
                    size_t total = dltotal > 0 ? static_cast<size_t>(dltotal) : 0;
                    size_t now   = dlnow   > 0 ? static_cast<size_t>(dlnow)   : 0;
                    progressCallback(now, total);
                    return true;
                }});
        }
        cpr::Response resp = session.Download(ofs);
        ofs.flush();
        ofs.close();

        if (resp.error.code == cpr::ErrorCode::OK && resp.status_code >= 200 && resp.status_code < 300)
        {
            return true;
        }

        if (attempt >= mMaxRetries || !shouldRetry(resp))
        {
            std::remove(filePath.c_str());
            return false;
        }

        if (auto ra = retryAfterSeconds( resp))
        {
            sleepFor(std::chrono::seconds(*ra));
        }
        else
        {
            sleepFor(calculateBackoff(attempt));
        }

    }

    return false;

}

bool Http::streamDownload(const std::string& url, const cpr::Header& headers, std::function<bool(const std::string&)> chunkProcessor) const
{

    cpr::Header hdr = mergeHeaders(mDefaultHeaders, headers);

    for (int attempt = 1; attempt <= std::max(mMaxRetries, 1); ++attempt)
    {
        cpr::Session s;
        s.SetUrl(url);
        s.SetHeader(hdr);
        s.SetTimeout(mTimeoutSeconds * 1000);
        if (mDefaultAuth)
        {
            s.SetAuth(*mDefaultAuth);
        }
        std::string buffer;
        s.SetWriteCallback(cpr::WriteCallback{
        [&buffer, &chunkProcessor](std::string data, intptr_t)
        {
             if (!data.empty())
             {
                 if (!chunkProcessor(data))
                 {
                     return false;
                 }
             }
            return true;
        }
        });

        cpr::Response res = s.Get();

        if (res.error.code == cpr::ErrorCode::OK && res.status_code >= 200 && res.status_code < 300) {
            return true;
        }

        if (attempt >= mMaxRetries || !shouldRetry(res)) {
            return false;
        }

        if (auto ra = retryAfterSeconds(res)) {
            sleepFor(std::chrono::seconds(*ra));
        } else {
            sleepFor(calculateBackoff(attempt));
        }
    }
    return false;

std::string csvEscape(std::string_view s)
{
    bool needsQuotes = false;
    std::string out;
    out.reserve(s.size() + 8);

    for (auto ch : s)
    {
        if (ch == '"' || ch == ',' || ch == '\n' || ch == '\r')
        {
            needsQuotes = true;
        }
        if (ch == '"')
        {
            out.push_back('"');
            out.push_back('"');
        }
        else
        {
            out.push_back(ch);
        }
    }
    if (needsQuotes)
    {
        return "\"" + out + "\"";
    }
    return out;
}

static std::string pgEscapeText(std::string_view text)
{
    // For postgres array literals, we need to eascape " and \ with backslashes
    std::string out;
    out.reserve(text.size() + 8);
    for (char ch: text)
    {
        if (ch == '"' || ch == '\\' )
        {
            out.push_back('\\');
        }
        out.push_back(ch);
    }
    return out;
}

std::string pgArrayText(const std::vector<std::string>& items)
{
    std::string out;
    out.push_back('{');
    for (size_t i = 0; i < items.size(); i++)
    {
        out.push_back('"');
        out+=pgEscapeText(items[i]);
        out.push_back('"');
        if (i + 1 < items.size())
        {
            out.push_back(',');
        }
    }
    out.push_back('}');
    return out;
}

std::optional<std::string> getString(simdjson::ondemand::object obj, std::string_view key)
{
    auto it = obj.find_field(key);
    if (it.error())
    {
        return std::nullopt;
    }
    return getString(it.value());
}

std::optional<std::string> getString(simdjson::ondemand::value val)
{
    if (val.type() == simdjson::ondemand::json_type::string)
    {
       auto svRes = val.get_string();
        if (svRes.error())
        {
            return std::nullopt;
        }
        return std::string(svRes.value());
    }

    try
    {
        auto s = simdjson::to_json_string(val);
        if (s.error())
        {
            return std::nullopt;
        }
        return std::string(s.value());
    }
    catch (...)
    {
        return std::nullopt;
    }
}

std::optional<int> getInt(simdjson::ondemand::object obj, std::string_view key)
{
    auto it = obj.find_field(key);
    if (it.error())
    {
        return std::nullopt;
    }
    int64_t v{};
    auto e = it.value().get_int64().get(v);
    if (e)
    {
        return std::nullopt;
    }
    if (v < std::numeric_limits<int>::min() || v > std::numeric_limits<int>::max())
    {
        return std::nullopt;
    }

    return static_cast<int>(v);
}

bool getBool(simdjson::ondemand::object obj, std::string_view key, bool fallback)
{
    auto it = obj.find_field(key);
    if (it.error())
    {
        return fallback;
    }
    bool b{};
    if (!it.value().get_bool().get(b))
    {
        return b;
    }
    return fallback;
}

std::vector<std::string> getStringArray(simdjson::ondemand::object obj, std::string_view key)
{
    std::vector<std::string> out;
    auto it = obj.find_field(key);
    if (it.error())
    {
        return out;
    }
    simdjson::ondemand::array arr;
    if (it.value().get_array().get(arr))
    {
        return out;
    }
    for (auto v: arr)
    {
        simdjson::ondemand::value val;
        if (v.get(val) == simdjson::SUCCESS)
        {
            if (auto s = getString(val))
            {
                out.push_back(std::move(*s));
            }
        }
    }
    return out;
}

std::string parseCompanyRest(const std::string& json, CompanyRecord& out)
{
    auto paddedString = simdjson::padded_string(json);
    simdjson::ondemand::parser parser;
    auto doc = parser.iterate(paddedString);
    simdjson::ondemand::object root;
    if (auto e = doc.get_object().get(root))
    {
        return "Invalid JSON (company doc)";
    }

    // These are required
    if (auto s = getString(root, "company_number"))
    {
        out.companyNumber = *s;
    }
    else
    {
        return "Missing Company Number";
    }

    if (auto s = getString(root, "company_name"))
    {
        out.name = *s;
    }
    else
    {
        out.name.clear();
    }

    if (auto s = getString(root, "company_status"))
    {
        out.status = *s;
    }
    else
    {
        out.status.clear();
    }


    out.sicCodes = getStringArray(root, "sic_codes");
    out.incorporationDate = getString(root, "date_of_creation");
    out.lastAccountsMadeUpTo = std::nullopt;
    out.accountType = std::nullopt;

    // accounts -> last_accounts_made_up_to, account_type if present
    if (auto accv = root.find_field("accounts"); !accv.error())
    {
        simdjson::ondemand::object acc;
        if (!accv.value().get_object().get(acc))
        {
            if (auto lastv = acc.find_field_unordered("last_accounts"); !lastv.error())
            {
                simdjson::ondemand::object last;
                if (!lastv.value().get_object().get(last))
                {
                    if (auto s = getString(last, "made_up_to"))
                    {
                        out.lastAccountsMadeUpTo = *s;
                    }
                    if (auto t = getString(last, "type"))
                    {
                        out.accountType = *t;
                    }
                }
            }
        }
    }

    // Registered office -> could later derive region from postcode elsewhere
    // Financial snapshot: REST company profile generally doesn't include full financials.
    // Leave turnover/employees/net_assets as nullopt for REST.

    return {};

}

std::string parseOfficersRest(const std::string& json, const std::string& companyNumber, std::vector<OfficerRecord>& out)
{
    auto paddedString = simdjson::padded_string(json);
    simdjson::ondemand::parser parser;
    auto doc = parser.iterate(paddedString);
    simdjson::ondemand::object root;
    if (auto e = doc.get_object().get(root))
    {
        return "Invalid JSON (officer root)";
    }

    simdjson::ondemand::array items;
    if (auto e = root.find_field("items").value().get_array().get(items))
    {
        // officer for a company might be genuinely be empty
        return {};
    }

    for (auto v: items)
    {
        simdjson::ondemand::object oi;
        if (v.get_object().get(oi))
        {
            continue;
        }

        OfficerRecord rec;
        rec.companyNumber = companyNumber;
        rec.name = getString(oi, "name").value_or("");
        rec.role = getString(oi, "officer_role").value_or("");

        rec.appointedOn = getString(oi, "appointed_on").value_or("");
        rec.resignedOn = getString(oi, "resigned_on");
        rec.isActive = !rec.resignedOn.has_value();

        // DOB Year
        if (auto dobv = oi.find_field_unordered("date_of_birth"); !dobv.error())
        {
            simdjson::ondemand::object dob;
            if (!dobv.value().get_object().get(dob))
            {
                if (auto y = getInt(dob, "year"))
                {
                    rec.dobYear = *y;
                }
            }
        }

        out.push_back(std::move(rec));
    }

    return {};
}

static std::string pscNameFromElements(simdjson::ondemand::object obj)
{
    std::vector<std::string> parts;
    auto addIfPresent = [&](std::string_view key)
    {
        if (auto s = getString(obj, key))
        {
            if (!s->empty())
            {
                parts.push_back(*s);
            }
        }
    };

    addIfPresent("title");
    addIfPresent("forename");
    addIfPresent("other_forenames");
    addIfPresent("surname");

    std::string name;
    for (size_t i = 0; i < parts.size(); ++i)
    {
        if (!name.empty())
        {
            name.push_back(' ');
        }
        name += parts[i];
    }
    return name;
}

std::string parsePSCRest(const std::string& json, const std::string& companyNumber, std::vector<PscRecord>& out)
{
    auto paddedString = simdjson::padded_string(json);
    simdjson::ondemand::parser parser;
    auto doc = parser.iterate(paddedString);
    simdjson::ondemand::object root;
    if (auto e = doc.get_object().get(root))
    {
        return "Invalid JSON (psc root)";
    }

    simdjson::ondemand::array items;
    if (auto e = root.find_field("items").value().get_array().get(items))
    {
        return {};
    }

    for (auto v : items)
    {
        simdjson::ondemand::object obj;
        if (v.get_object().get(obj))
        {
            continue;
        }

        PscRecord rec;
        rec.companyNumber = companyNumber;
        rec.name = getString(obj, "name").value_or("");

        if (rec.name.value().empty())
        {
            if (auto nameElementsVal = obj.find_field_unordered("name_elements"); !nameElementsVal.error())
            {
                simdjson::ondemand::object nameElements;
                if (!nameElementsVal.value().get_object().get(nameElements))
                {
                    rec.name = pscNameFromElements(nameElements);
                }
            }
        }

        rec.kind = getString(obj, "kind").value_or("");
        rec.naturesOfControl = getStringArray(obj, "natures_of_control");
        rec.description         = getString(obj, "description");
        rec.nationality         = getString(obj, "nationality");
        rec.countryOfResidence  = getString(obj, "country_of_residence");
        rec.notifiedOn          = getString(obj, "notified_on");
        rec.ceasedOn            = getString(obj, "ceased_on");
        rec.ceased              = getBool(obj, "ceased");
        rec.isSanctioned        = getBool(obj, "is_sanctioned");
        rec.etag                = getString(obj, "etag");

        if (auto dobV = obj.find_field_unordered("date_of_birth"); !dobV.error())
        {
            simdjson::ondemand::object dob;
            if (!dobV.value().get_object().get(dob))
            {
                rec.dobYear = getInt(dob, "year");
                rec.dobMonth = getInt(dob, "month");
            }
        }

        // Nested JSON Blobs
        if (auto aV = obj.find_field_unordered("address"); !aV.error())
        {
            rec.addressJson = objToJson(aV.value());
        }

        if (auto paV = obj.find_field_unordered("principal_office_address"); !paV.error())
        {
            rec.principalOfficeAddressJson = objToJson(paV.value());
        }

        if (auto idV = obj.find_field_unordered("identification"); !idV.error())
        {
            rec.identificationJson = objToJson(idV.value());
        }

        if (auto ivV = obj.find_field_unordered("identity_verification_details"); !ivV.error())
        {
            rec.identityVerificationJson = objToJson(ivV.value());
        }

        if (auto linksV = obj.find_field_unordered("links"); !linksV.error())
        {
            simdjson::ondemand::object links;
            if (!linksV.value().get_object().get(links))
            {
                rec.linkSelf = getString(links, "self");
                rec.linkStatement = getString(links, "statement");
            }
        }
        out.push_back(std::move(rec));
    }
    return {};
}


bool parseCompanyNdjsonLine(std::string_view line, CompanyRecord& out)
{
    if (line.empty())
    {
        return false;
    }

    simdjson::ondemand::parser parser;
    simdjson::padded_string padded(line);
    auto doc = parser.iterate(padded);
    simdjson::ondemand::object root;
    if (doc.get_object().get(root))
    {
        return false;
    }

    auto companyNumber = getString(root, "company_number");
    if (!companyNumber)
    {
        return false;
    }
    out.companyNumber = *companyNumber;

    auto companyName = getString(root, "company_name");
    out.name = companyName.value_or("");
    out.status = getString(root, "company_status").value_or("");
    out.sicCodes = getStringArray(root, "sic_codes");

    auto incorporation = getString(root, "date_of_creation");
    if (!incorporation)
    {
        incorporation = getString(root, "incorporation_date");
    }
    out.incorporationDate = incorporation;

    if (auto accv = root.find_field("accounts"); !accv.error())
    {
        simdjson::ondemand::object accounts;
        if (!accv.value().get_object().get(accounts))
        {
            if (auto lastv = accounts.find_field_unordered("last_accounts"); !lastv.error())
            {
                simdjson::ondemand::object last;
                if (!lastv.value().get_object().get(last))
                {
                    if (auto s = getString(last, "made_up_to"))
                    {
                        out.lastAccountsMadeUpTo = *s;
                    }
                    if (auto t = getString(last, "type"))
                    {
                        out.accountType = *t;
                    }
                }
            }
        }
    }

    out.turnover       = getString(root, "turnover");
    out.employees      = getString(root, "employees");
    out.netAssets      = getString(root, "net_assets");
    out.currentAssets  = getString(root, "current_assets");
    out.creditors      = getString(root, "creditors");
    out.cash           = getString(root, "cash");

    return true;
}

size_t LineAccumulator::feed(std::string_view chunk, const std::function<void(std::string_view)>& onLine) {
    size_t emitted = 0;
    buffer.append(chunk.data(), chunk.size());
    size_t pos = 0;
    while (true)
    {
        auto nl = buffer.find('\n', pos);
        if (nl == std::string::npos)
        {
            // keep remainder in buffer
            buffer.erase(0, pos);
            return emitted;
        }
        // line without trailing '\r' (handle CRLF)
        size_t line_end = (nl > 0 && buffer[nl - 1] == '\r') ? nl - 1 : nl;
        onLine(std::string_view(buffer.data() + pos, line_end - pos));
        emitted++;
        pos = nl + 1;
    }
}

void LineAccumulator::flush(const std::function<void(std::string_view)>& onLine) {
    if (!buffer.empty())
    {
        onLine(std::string_view(buffer));
        buffer.clear();
    }
}

      << csvEscape(optInt(r.dobYear)) << ','

    bool resolveConfigPath(std::filesystem::path& resolvedPath, std::string& errorOut)
    {
        static constexpr const char* kConfigEnvVar = "CHAPP_INGEST_CONFIG";

    bool tryAcquireProcessAdvisoryLock(pqxx::connection& conn)
    {
        pqxx::work w(conn);
        const auto row = w.exec("SELECT pg_try_advisory_lock($1)", pqxx::params{kProcessAdvisoryLockKey}).one_row();
        const bool acquired = row[0].as<bool>();
        w.commit();
        return acquired;
    }

    void releaseProcessAdvisoryLock(pqxx::connection& conn)
    {
        try
        {
            pqxx::work w(conn);
            w.exec("SELECT pg_advisory_unlock($1)", pqxx::params{kProcessAdvisoryLockKey});
            w.commit();
        }

#pragma once

#include <libpq-fe.h>
#include <stdexcept>
#include <string>

namespace pg
{

    class CopySession
    {
    public:
        explicit CopySession(pg_conn* conn, const std::string& copySql) : mConn(conn)
        {
            if (!mConn || PQstatus(mConn) != CONNECTION_OK)
            {
                throw std::runtime_error("CopySession: invalid PGConn failed");
            }
            PGresult *r = PQexec(mConn, copySql.c_str());
            if (PQresultStatus(r) != PGRES_COPY_IN)
            {
                std::string err = PQerrorMessage(mConn);
                PQclear(r);
                throw std::runtime_error("COPY START Failed" + err);
            }
            PQclear(r);
            mStarted = true;
        }

        CopySession(const CopySession&) = delete;
        CopySession& operator=(const CopySession&) = delete;

        ~CopySession()
        {
            if (mStarted && !mFinished)
            {
                PQputCopyEnd(mConn, "aborted");
                while (PGresult* r = PQgetResult(mConn))
                {
                    PQclear(r);
                }
            }
        }

        void putCsvLine(const std::string& csvLine) const
        {
            if (PQputCopyData(mConn, csvLine.c_str(), static_cast<int>(csvLine.size())) != 1)
            {
                throw std::runtime_error("PQputCopyData failed");
            }
            if (PQputCopyData(mConn, "\n", 1) != 1)
            {
                throw std::runtime_error("PQputCopyData new line failed");
            }
        }

        void finish()
        {
            if (!mStarted || mFinished)
            {
                return;
            }

            if (PQputCopyEnd(mConn, nullptr)!= 1)
            {
                throw std::runtime_error("PQputCopyEnd failed");
            }

            PGresult* done = PQgetResult(mConn);
            if (PQresultStatus(done) != PGRES_COMMAND_OK)
            {
                std::string err = PQerrorMessage(mConn);
                PQclear(done);
                throw std::runtime_error("PQputCopyEnd failed: " + err);
            }
            PQclear(done);
            mFinished = true;
        }

        private:

        pg_conn* mConn{nullptr};
        bool mStarted{false};
        bool mFinished{false};
    };
}

        std::string monthNameToLower(std::string value)
        {
            for (char& c : value)
            {
                c = static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
            }
            return value;
        }

        ParsedMonthlyKey parseMonthlyKeyOrThrow(const std::string& key)
        {
            static constexpr int YEAR_LENGTH = 4;
            static constexpr int MIN_KEY_LENGTH = YEAR_LENGTH + 1;  // Month name (min 1 char) + year

            if (key.size() < MIN_KEY_LENGTH)
            {
                throw std::runtime_error("Invalid monthly key: " + key);
            }

            const std::string yearPart = key.substr(key.size() - YEAR_LENGTH);
            for (const char c : yearPart)
            {
                if (!std::isdigit(static_cast<unsigned char>(c)))
                {
                    throw std::runtime_error("Invalid monthly key year: " + key);
                }
            }

            static const std::map<std::string, int> monthMap{
                {"january", 1}, {"february", 2}, {"march", 3}, {"april", 4},
                {"may", 5}, {"june", 6}, {"july", 7}, {"august", 8},
                {"september", 9}, {"october", 10}, {"november", 11}, {"december", 12}
            };

            ParsedMonthlyKey parsed{};
            parsed.year = std::stoi(yearPart);

            std::string monthPart = key.substr(0, key.size() - YEAR_LENGTH);
            const auto toPos = monthPart.find("To");
            if (toPos != std::string::npos && toPos + 2 < monthPart.size())
            {
                monthPart = monthPart.substr(toPos + 2);
            }
            monthPart = monthNameToLower(monthPart);
            const auto it = monthMap.find(monthPart);
            if (it == monthMap.end())
            {
                throw std::runtime_error("Invalid month name in key: " + key);
            }

            parsed.month = it->second;
            return parsed;
        }

            if (overrideDate)

                return fmt::format("{:04}-{:02}-{:02}",
                                   static_cast<int>(overrideDate->year()),
                                   static_cast<unsigned>(overrideDate->month()),
                                   static_cast<unsigned>(overrideDate->day()));
            }

            const auto now = std::chrono::floor<std::chrono::days>(std::chrono::system_clock::now());
            const std::chrono::year_month_day ymd{now};
            return fmt::format("{:04}-{:02}-01",
                               static_cast<int>(ymd.year()),
                               static_cast<unsigned>(ymd.month()));
        }

        JobRunResult ingestDownloadedAccountsFiles(const std::string& sourceName,
                                                   const bulk::AccountsDownloadResult& result,
                                                   pqxx::connection& db,
                                                   bulk::AccountsIxbrlIngestService& ingestService,
                                                   bulk::AccountsIxbrlIngestOptions ingestOptions,
                                                   PGconn* rawConn,
                                                   std::size_t* outIngested)
        {
            std::size_t ingested = 0;
            std::size_t skipped = 0;
            std::size_t failed = 0;
            bulk::IxbrlIngestStats aggregate{};

            for (const auto& file : result.files)
            {
                if (!file.downloaded)
                {
                    failed++;
                    continue;
                }

                if (!beginArtifactAttempt(db, sourceName, file.link.key, file.link.url))
                {
                    skipped++;
                    continue;
                }

                try
                {
                    bulk::IxbrlIngestStats fileStats{};
                    const bool ok = ingestService.ingestZipFile(file.localPath, db, rawConn, ingestOptions, &fileStats);
                    aggregate.documentsTotal += fileStats.documentsTotal;
                    aggregate.documentsOk += fileStats.documentsOk;
                    aggregate.documentsSkipped += fileStats.documentsSkipped;
                    aggregate.documentsBad += fileStats.documentsBad;
                    aggregate.factsInserted += fileStats.factsInserted;
                    aggregate.contextsInserted += fileStats.contextsInserted;
                    aggregate.unitsInserted += fileStats.unitsInserted;

                    if (ok)
                    {
                        finishArtifactAttempt(db, sourceName, file.link.key, file.link.url, true, "");
                        ingested++;
                    }
                    else
                    {
                        finishArtifactAttempt(db, sourceName, file.link.key, file.link.url, false, "iXBRL ingest reported failures");
                        failed++;
                    }
                }
                catch (const std::exception& e)
                {
                    finishArtifactAttempt(db, sourceName, file.link.key, file.link.url, false, e.what());
                    failed++;
                }
            }

            if (outIngested)
            {
                *outIngested = ingested;
            }

            JobRunResult resultOut;
            resultOut.success = failed == 0;
            resultOut.message = fmt::format(
                "accounts ingest source={} ingested={} skipped={} failed={} documents_ok={} documents_bad={} facts={}",
                sourceName,
                ingested,
                skipped,
                failed,
                aggregate.documentsOk,
                aggregate.documentsBad,
                aggregate.factsInserted
            );
            return resultOut;
        }
    }

    JobRunResult runAccountsBackfillJob(const IngestConfig& config,
                                        Http& http,
                                        const ImportStage,
                                        const AccountsBackfillJobOptions& options)
    {
        pqxx::connection db{config.getDbConnectionString()};
        pg::ensureMinimalSchema(db);
        pg::Conn rawConn{config.getDbConnectionString()};
        if (!rawConn.valid())
        {
            return {false, "accounts backfill DB connection is invalid"};
        }

        bulk::AccountsIxbrlIngestService ingestService(config);
        bulk::AccountsIxbrlIngestOptions ingestOptions{};
        ingestOptions.verbose = options.verbose;
        bulk::AccountsDownloadService downloadService(config.bulkDownloadDir, http);

        std::size_t totalProcessed = 0;
        std::size_t totalFailed = 0;
        std::size_t totalSkipped = 0;

        const std::array<std::pair<bulk::AccountsSource, std::size_t>, 2> sources{
            std::pair{bulk::AccountsSource::Monthly, options.maxMonthlyFilesPerRun},
            std::pair{bulk::AccountsSource::HistoricMonthly, options.maxHistoricFilesPerRun}
        };

        for (const auto& sourceItem : sources)
        {
            const auto source = sourceItem.first;
            const auto maxPerRun = sourceItem.second;
            if (maxPerRun == 0)
            {
                continue;
            }

            bulk::AccountsDownloadOptions queryOptions{};
            queryOptions.source = source;
            queryOptions.verbose = options.verbose;
            queryOptions.keepFiles = options.keepFiles;
            queryOptions.fallbackToLatest = false;

            const std::string listingUrl = (source == bulk::AccountsSource::Monthly)
                ? "https://download.companieshouse.gov.uk/en_monthlyaccountsdata.html"
                : "https://download.companieshouse.gov.uk/historicmonthlyaccountsdata.html";
            std::string html;
            if (!http.getText(listingUrl, {{"Accept", "text/html"}}, html) || html.empty())
            {
                totalFailed++;
                continue;
            }

            auto links = bulk::extractMonthlyAccountsLinksFromHtml(html, source);
            std::sort(links.begin(), links.end(), [](const bulk::AccountsZipLink& lhs, const bulk::AccountsZipLink& rhs)
            {
                const auto a = parseMonthlyKeyOrThrow(lhs.key);
                const auto b = parseMonthlyKeyOrThrow(rhs.key);
                if (a.year != b.year)
                {
                    return a.year < b.year;
                }
                if (a.month != b.month)
                {
                    return a.month < b.month;
                }
                return lhs.key < rhs.key;
            });

            std::size_t processedForSource = 0;
            for (const auto& link : links)
            {
                if (processedForSource >= maxPerRun)
                {
                    break;
                }

                const std::string sourceName = sourceToArtifactPrefix(source);
                if (!beginArtifactAttempt(db, sourceName, link.key, link.url))
                {
                    totalSkipped++;
                    continue;
                }

                queryOptions.monthlyKey = link.key;
                bulk::AccountsDownloadResult downloadResult;
                try
                {
                    downloadResult = downloadService.download(queryOptions);
                    if (!downloadResult.ok())
                    {
                        finishArtifactAttempt(db, sourceName, link.key, link.url, false, "download failed");
                        totalFailed++;
                        continue;
                    }
                }
                catch (const std::exception& e)
                {
                    finishArtifactAttempt(db, sourceName, link.key, link.url, false, e.what());
                    totalFailed++;
                    continue;
                }

                bool ingestOk = true;
                for (const auto& file : downloadResult.files)
                {
                    if (!file.downloaded)
                    {
                        ingestOk = false;
                        continue;
                    }
                    try
                    {
                        bulk::IxbrlIngestStats fileStats{};
                        const bool ok = ingestService.ingestZipFile(file.localPath, db, rawConn.get(), ingestOptions, &fileStats);
                        if (!ok)
                        {
                            ingestOk = false;
                            finishArtifactAttempt(db, sourceName, link.key, link.url, false, "iXBRL ingest reported failures");
                            break;
                        }
                    }
                    catch (const std::exception& e)
                    {
                        ingestOk = false;
                        finishArtifactAttempt(db, sourceName, link.key, link.url, false, e.what());
                        break;
                    }
                }

                if (ingestOk)
                {
                    finishArtifactAttempt(db, sourceName, link.key, link.url, true, "");
                    processedForSource++;
                    totalProcessed++;
                }
                else
                {
                    totalFailed++;
                }
            }
        }

        JobRunResult result{};
        result.success = totalFailed == 0;
        result.message = fmt::format("accounts backfill processed={} skipped={} failed={}",
                                     totalProcessed,
                                     totalSkipped,
                                     totalFailed);
        return result;
    }

    JobRunResult runPscWeeklyJob(const IngestConfig& config,
                                 Http& http,
                                 const ImportStage stage,
                                 const PscWeeklyJobOptions& options)
    {
        if (stage == ImportStage::Merge)
        {
            return {false, "PSC weekly job does not support merge stage"};
        }

        pqxx::connection db{config.getDbConnectionString()};
        pg::ensureMinimalSchema(db);
        pg::Conn rawConn{config.getDbConnectionString()};
        if (!rawConn.valid())
        {
            return {false, "PSC weekly DB connection is invalid"};
        }

        std::optional<std::string> snapshotDate = options.dateOverride
            ? std::optional<std::string>{fmt::format("{:04}-{:02}-{:02}",
                                                     static_cast<int>(options.dateOverride->year()),
                                                     static_cast<unsigned>(options.dateOverride->month()),
                                                     static_cast<unsigned>(options.dateOverride->day()))}
            : latestPscDate(http, options.mode);
        const std::string snapshotKey = snapshotDate.value_or("unknown");
        const std::string artifactUrl = "psc-snapshot-" + snapshotKey;
        if (!beginArtifactAttempt(db, "psc_weekly", snapshotKey, artifactUrl))
        {
            return {true, "PSC weekly snapshot already ingested"};
        }

        bulk::PscIngestService pscService(config, http);
        bulk::SeedStats stats{};
        std::optional<std::string> resolvedDate;
        const std::string targetTable = (stage == ImportStage::StageSwap) ? "psc_persons_stage" : "psc_persons";

        try
        {
            const bool ok = pscService.ingestDailySnapshot(rawConn.get(),
                                                           options.dateOverride,
                                                           options.mode,
                                                           &stats,
                                                           targetTable,
                                                           true,
                                                           &resolvedDate);
            if (!ok)
            {
                finishArtifactAttempt(db, "psc_weekly", snapshotKey, artifactUrl, false, "PSC ingest failed");
                return {false, "PSC ingest failed"};
            }

            if (stage == ImportStage::StageSwap)
            {
                pqxx::work w(db);
                w.exec("TRUNCATE TABLE psc_persons").no_rows();
                w.exec(
                    "INSERT INTO psc_persons("
                    "company_number,psc_key,name,kind,natures_of_control,notified_on,ceased_on,ceased,is_sanctioned,"
                    "nationality,country_of_residence,dob_year,dob_month,description,address,principal_office_address,"
                    "identification,identity_verification,link_self,link_statement,etag,raw_json,updated_at"
                    ") "
                    "SELECT company_number,psc_key,name,kind,natures_of_control,notified_on,ceased_on,ceased,is_sanctioned,"
                    "nationality,country_of_residence,dob_year,dob_month,description,address,principal_office_address,"
                    "identification,identity_verification,link_self,link_statement,etag,raw_json,updated_at "
                    "FROM psc_persons_stage").no_rows();
                w.commit();
            }

            const std::string snapshotForHistory = resolvedDate.value_or(snapshotKey);
            if (snapshotForHistory != "unknown")
            {
                insertPscSnapshotHistory(db, snapshotForHistory);
            }

            finishArtifactAttempt(db, "psc_weekly", snapshotKey, artifactUrl, true, "");
            return {
                true,
                fmt::format("psc weekly loaded snapshot={} lines_ok={} lines_bad={}",
                            snapshotForHistory,
                            stats.linesOk,
                            stats.linesBad)
            };
        }
        catch (const std::exception& e)
        {
            finishArtifactAttempt(db, "psc_weekly", snapshotKey, artifactUrl, false, e.what());
            return {false, e.what()};
        }
    }

    JobRunResult runBulkCompanyDataJob(const IngestConfig& config,
                                       Http& http,
                                       const ImportStage stage,
                                       const BulkCompanyJobOptions& options)
    {
        pqxx::connection db{config.getDbConnectionString()};
        pg::ensureMinimalSchema(db);
        pg::Conn rawConn{config.getDbConnectionString()};
        if (!rawConn.valid())
        {
            return {false, "bulk company DB connection is invalid"};
        }

        const std::string monthKey = ymdKey(options.dateOverride);
        const std::string artifactUrl = "bulk-companies-" + monthKey;
        if (!beginArtifactAttempt(db, "bulk_company_data", monthKey, artifactUrl))
        {
            return {true, "bulk company snapshot already ingested"};
        }

        bulk::BulkIngestService service(config, http);
        bulk::SeedStats stats{};
        const std::string stageTable = (stage == ImportStage::Truncate) ? "companies" : "companies_stage";

        try
        {
            const bool loaded = service.ingestMonthlyFile(rawConn.get(),
                                                          options.dateOverride,
                                                          &stats,
                                                          stageTable,
                                                          true);
            if (!loaded)
            {
                finishArtifactAttempt(db, "bulk_company_data", monthKey, artifactUrl, false, "bulk load failed");
                return {false, "bulk load failed"};
            }

            if (stage == ImportStage::Merge)
            {
                pqxx::work w(db);
                w.exec(
                    "INSERT INTO companies("
                    "company_number,name,status,sic_codes,incorporation_date,region,last_accounts_made_up_to,account_type,updated_at"
                    ") "
                    "SELECT company_number,name,status,sic_codes,incorporation_date,region,last_accounts_made_up_to,account_type,now() "
                    "FROM companies_stage "
                    "ON CONFLICT (company_number) DO UPDATE SET "
                    "name=EXCLUDED.name, status=EXCLUDED.status, sic_codes=EXCLUDED.sic_codes, "
                    "incorporation_date=EXCLUDED.incorporation_date, region=EXCLUDED.region, "
                    "last_accounts_made_up_to=EXCLUDED.last_accounts_made_up_to, account_type=EXCLUDED.account_type, "
                    "updated_at=now() "
                    "WHERE "
                    "(companies.name,companies.status,companies.sic_codes,companies.incorporation_date,companies.region,"
                    "companies.last_accounts_made_up_to,companies.account_type) "
                    "IS DISTINCT FROM "
                    "(EXCLUDED.name,EXCLUDED.status,EXCLUDED.sic_codes,EXCLUDED.incorporation_date,EXCLUDED.region,"
                    "EXCLUDED.last_accounts_made_up_to,EXCLUDED.account_type)"
                ).no_rows();
                w.commit();
            }
            else if (stage == ImportStage::StageSwap)
            {
                pqxx::work w(db);
                w.exec("TRUNCATE TABLE companies").no_rows();
                w.exec(
                    "INSERT INTO companies("
                    "company_number,name,status,sic_codes,incorporation_date,region,last_accounts_made_up_to,account_type,updated_at"
                    ") "
                    "SELECT company_number,name,status,sic_codes,incorporation_date,region,last_accounts_made_up_to,account_type,now() "
                    "FROM companies_stage"
                ).no_rows();
                w.commit();
            }

            finishArtifactAttempt(db, "bulk_company_data", monthKey, artifactUrl, true, "");
            return {
                true,
                fmt::format("bulk company load complete month={} lines_ok={} lines_bad={}",
                            monthKey,
                            stats.linesOk,
                            stats.linesBad)
            };
        }
        catch (const std::exception& e)
        {
            finishArtifactAttempt(db, "bulk_company_data", monthKey, artifactUrl, false, e.what());
            return {false, e.what()};
        }
    }

namespace bulk

    static std::string fileNameFromUrlOrThrow(const std::string& url)
    {
        const auto slash = url.find_last_of('/');
        if (slash == std::string::npos || slash + 1 >= url.size())
        {
            throw std::runtime_error("Invalid URL, cannot derive filename" + url);
        }

        return url.substr(slash + 1);
    }

    static bool isValidDailyKey(const std::string& s)
    {
        // YYYY-MM-DD
        if (s.size() != 10)
        {
            return false;
        }
        for (int i = 0; i < 10; ++i)
        {
            if (i == 4 || i == 7)
            {
                if (s[i] != '-')
                {return false;
                }
            }
            else
            {
                if(!std::isdigit(static_cast<unsigned char>(s[i])))
                {
                    return false;
                }
            }
        }
        return true;
    }

    static int monthIndexOrThrow(std::string& monthName)
    {
        for (auto&c : monthName)
        {
           c =  static_cast<char>(std::tolower(static_cast<unsigned char>(c)));
        }

        static const std::map<std::string, int> months{
            {"january",1},{"february",2},{"march",3},{"april",4},{"may",5},{"june",6},
            {"july",7},{"august",8},{"september",9},{"october",10},{"november",11},{"december",12}
        };

        auto it = months.find(monthName);
        if (it == months.end())
        {
            throw std::runtime_error("Unknown month name: " + monthName);
        }

        return it->second;
    }

    static MonthlyKeyParts parseMonthlyKeyOrThrow(const std::string& key)
    {
        // Supported:
        // - MonthNameYYYY            e.g. "October2025"
        // - JanuaryToDecemberYYYY    e.g. "JanuaryToDecember2008" (historic year pack)
        //
        // Parsing strategy:
        // - last 4 chars must be digits (year)
        // - prefix is month or range
        if (key.size() < 5)
        {
            throw std::runtime_error("Invalid monthly key: " + key);
        }

        const std::string yearString = key.substr(key.size() - 4);
        for (char c : yearString)
        {
            if (!std::isdigit(static_cast<unsigned char>(c)))
            {
                throw std::runtime_error("Invalid monthly key year: " + key);
            }
        }

        MonthlyKeyParts out;
        out.year = std::stoi(yearString);

        std::string prefix = key.substr(0, key.size() - 4);

        // Detect YearPack: <Month>To<Month>
        // We treat JanuaryToDecemberYYYY as month=12 for ordering (end-of-year snapshot).
        const auto toPos = prefix.find("To");
        if (toPos != std::string::npos && toPos > 0 && toPos + 2 < prefix.size())
        {
            out.yearPack = true;
            std::string endMonth = prefix.substr(toPos + 2); // after "To"
            out.month = monthIndexOrThrow(endMonth);
            return out;
        }

        out.month = monthIndexOrThrow(prefix);
        return out;
    }

    static void requireSourceOrThrow(const std::vector<AccountsZipLink>& links, AccountsSource expected)
    {
        for (const auto& l : links)
        {
            if (l.source != expected)
            {
                throw std::runtime_error("Link source mismatch in plan build (mixed sources vector)");
            }
        }
    }

    AccountsDownloadPlan buildDailyAccountsPlan(std::vector<AccountsZipLink> links, const AccountsDownloadOptions& opt)
    {
        AccountsDownloadPlan plan;
        plan.source = opt.source;

        // Validate daily keys and drop invalid ones or throw
        for (const auto& l : links)
        {
            if (!isValidDailyKey(l.key))
            {
                throw std::runtime_error("Invalid daily key: " + l.key);
            }
        }

        // Exact date request
        if (opt.dailyDate)
        {
            if (!isValidDailyKey(*opt.dailyDate))
            {
                throw std::runtime_error("Invalid requested daily date: " + *opt.dailyDate);
            }

            auto it = std::find_if(links.begin(), links.end(),
                                   [&](const AccountsZipLink& l)
                                   { return l.key == *opt.dailyDate; });
            if (it != links.end())
            {
                plan.links.push_back(*it);
                return plan;
            }

            if (!opt.fallbackToLatest)
            {
                throw std::runtime_error("Requested daily date not found: " + *opt.dailyDate);
            }
            // fallback => continue to "latest N"
        }

        int n = opt.latestDailyCount;
        if (n < 1)
        {
            throw std::runtime_error("latestDailyCount must be >= 1");
        }

        std::sort(links.begin(), links.end(),
                  [](const AccountsZipLink& a, const AccountsZipLink& b)
                  {
                      return a.key > b.key;
                  });

        if (static_cast<int>(links.size()) < n) n = static_cast<int>(links.size());

        plan.links.assign(links.begin(), links.begin() + n);
        return plan;
    }

    AccountsDownloadPlan buildMonthlyAccountsPlan(std::vector<AccountsZipLink> links, const AccountsDownloadOptions& opt)
    {
        AccountsDownloadPlan plan;
        plan.source = opt.source;

        // Exact key request
        if (opt.monthlyKey)
        {
            auto it = std::find_if(links.begin(), links.end(),
                                   [&](const AccountsZipLink& l) { return l.key == *opt.monthlyKey; });

            if (it != links.end())
            {
                plan.links.push_back(*it);
                return plan;
            }

            if (!opt.fallbackToLatest)
            {
                throw std::runtime_error("Requested monthly key not found: " + *opt.monthlyKey);
            }
            // fallback to latest
        }

        // Latest by (year, month)
        std::sort(links.begin(), links.end(),
                  [](const AccountsZipLink& a, const AccountsZipLink& b)
                  {
                      const auto pa = parseMonthlyKeyOrThrow(a.key);
                      const auto pb = parseMonthlyKeyOrThrow(b.key);

                      if (pa.year != pb.year) return pa.year > pb.year;
                      return pa.month > pb.month;
                  });

        plan.links.push_back(links.front());
        return plan;
    }

    AccountsDownloadPlan buildAccountsDownloadPlanOrThrow(std::vector<AccountsZipLink> links, const AccountsDownloadOptions& opt)
    {
        if (links.empty())
        {
            throw std::runtime_error("No accounts links provided to planner");
        }

        requireSourceOrThrow(links, opt.source);

        if (opt.source == AccountsSource::Daily)
        {
            return buildDailyAccountsPlan(std::move(links), opt);
        }
        return buildMonthlyAccountsPlan(std::move(links), opt);
    }

    std::string AccountsDownloadService::listingUrlFor(AccountsSource source) const
    {
        switch (source)
        {
            case AccountsSource::Daily:
                return "https://download.companieshouse.gov.uk/en_accountsdata.html";
            case AccountsSource::Monthly:
                return "https://download.companieshouse.gov.uk/en_monthlyaccountsdata.html";
            case AccountsSource::HistoricMonthly:
                return "https://download.companieshouse.gov.uk/historicmonthlyaccountsdata.html";
        }

        throw std::runtime_error("Unknown source specified:");
    }

    std::vector<AccountsZipLink> AccountsDownloadService::fetchAndExtract(AccountsSource source) const
    {
        const std::string url = listingUrlFor(source);
        std::string html;

        const bool got = mHttp.getText(url, {{"Accept", "text/html"}}, html);
        if (!got || html.empty())
        {
            throw std::runtime_error("Failed to fetch listing HTML: " + url);
        }

        if (source == AccountsSource::Daily)
        {
            return extractDailyAccountsLinksFromHtml(html);
        }
        return extractMonthlyAccountsLinksFromHtml(html, source);
    }

    std::filesystem::path AccountsDownloadService::makeLocalPath(const AccountsZipLink& link) const
    {
        const std::string filename = !link.filename.empty() ? link.filename : fileNameFromUrlOrThrow(link.url);
        return mDownloadDirectory / filename;
    }

    AccountsDownloadResult AccountsDownloadService::download(const AccountsDownloadOptions& options)
    {
        AccountsDownloadResult result;

        auto links = fetchAndExtract(options.source);
        auto plan = buildAccountsDownloadPlanOrThrow(std::move(links), options);

        if (plan.links.empty())
        {
            throw std::runtime_error("Plan contains no links");
        }

        for (const auto& chosen : plan.links)
        {
            AccountsDownloadedFile file{};
            file.link = chosen;
            file.localPath = makeLocalPath(chosen);

            if (options.verbose)
            {
                PLOGI << "[AccountsDownload] Source: "
                      << (options.source == AccountsSource::Daily ? "Daily" :
                          options.source == AccountsSource::Monthly ? "Monthly" : "Historic");
                PLOGI << "[AccountsDownload] URL: " << chosen.url;
                PLOGI << "[AccountsDownload] Out: " << file.localPath.string();
            }

            const bool ok = mHttp.downloadFile(chosen.url, file.localPath.string(), {},
                [](size_t now, size_t total)
                {
                    if (total > 0 && (now %(1024 * 1024 * 25) == 0))
                    {
                        PLOGD << "[AccountsDownload] Downloaded "
                              << (now / (1024 * 1024)) << "MB / "
                              << (total / (1024 * 1024)) << "MB";
                    }
                });

            if (!ok)
            {
                file.downloaded = false;
                file.error = "download failed";
                std::error_code ec;
                std::filesystem::remove(file.localPath, ec);
            }
            else
            {
                file.downloaded = true;
                if (!options.keepFiles)
                {
                    std::error_code ec;
                    std::filesystem::remove(file.localPath, ec);
                }
            }

            result.files.push_back(std::move(file));
        }

        return result;
    }

gz::Inflater::Inflater(HeaderMode mode, size_t outChunk) : mMode(mode), mOutBuffer(outChunk)
{
    std::memset(&mZs, 0, sizeof(mZs));
    int rc = inflateInit2(&mZs, static_cast<int>(mode));
    if (rc != Z_OK)
    {
        throwZlibError("inflateInit2", rc);
    }
    mInit = true;
}

gz::Inflater::~Inflater()
{
    endStream();
}

void gz::Inflater::ensureInit()
{
    if (!mInit)
    {
        int rc = inflateInit2(&mZs, static_cast<int>(mMode));
        if (rc != Z_OK)
        {
            throwZlibError("inflateInit2", rc);
        }
        mInit = true;
        mEnded = false;
    }
}

void gz::Inflater::endStream()
{
    if (mInit)
    {
        inflateEnd(&mZs);
        mInit = false;
    }
}

void gz::Inflater::throwZlibError(const char* where, int code) const
{
    const char * msg = mZs.msg ? mZs.msg : "";
    throw std::runtime_error(std::string(where) + " failed: " + std::to_string(code) + (msg[0] ? std::string(" (") + msg + ")" : ""));
}

size_t gz::Inflater::feed(const uint8_t* data, size_t len, const std::function<void(const uint8_t*, size_t)>& onOut)
{
    if (mEnded)
    {
        throw std::runtime_error("Inflater::feed after stream ended; call reset() first");
    }
    ensureInit();
    mZs.next_in = const_cast<Bytef*>(reinterpret_cast<const Bytef*>(data));
    mZs.avail_in = static_cast<uInt>(len);
    size_t produced = 0;

    while (mZs.avail_in > 0)
    {
        mZs.next_out = mOutBuffer.data();
        mZs.avail_out = static_cast<uInt>(mOutBuffer.size());
        int rc = inflate(&mZs, Z_NO_FLUSH);
        if (rc == Z_STREAM_END)
        {
            size_t have = mOutBuffer.size() - mZs.avail_out;
            if (have)
            {
                onOut(mOutBuffer.data(), have);
                produced += have;
            }
            mEnded = true;
            break;
        }

        else if (rc ==Z_OK)
        {
            size_t have = mOutBuffer.size() - mZs.avail_out;
            if (have)
            {
                onOut(mOutBuffer.data(), have);
                produced += have;
            }
        }
        else
        {
            throwZlibError("inflate", rc);
        }
    }

    return produced;
}

size_t gz::Inflater::finish(const std::function<void(const uint8_t*, size_t)>& onOut)
{
    ensureInit();
    if (mEnded)
    {
        return 0;
    }

    size_t produced = 0;
    int rc = Z_OK;
    do
    {
        mZs.next_out = mOutBuffer.data();
        mZs.avail_out = static_cast<uInt>(mOutBuffer.size());
        rc = inflate(&mZs, Z_FINISH);
        if (rc == Z_STREAM_END || rc == Z_OK)
        {
            size_t have = mOutBuffer.size() - mZs.avail_out;
            if (have)
            {
                onOut(mOutBuffer.data(), have);
                produced += have;
            }
        }

        else if (rc == Z_BUF_ERROR)
        {
            break;
        }
        else
        {
            throwZlibError("inflate(ZFINISH)", rc);
        }
    }
    while (rc != Z_STREAM_END);

    mEnded = true;
    return produced;

}

void gz::Inflater::reset(HeaderMode headerMode)
{
    endStream();
    mMode = headerMode;
    std::memset(&mZs, 0, sizeof(mZs));
    int rc = inflateInit2(&mZs, static_cast<int>(mMode));
    if (rc != Z_OK)
    {
        throwZlibError("inflateInit2(reset)", rc);
    }
    mInit = true;
    mEnded = false;
}

    zip_file_t* openTargetEntry(zip_t* archive, const ZipSeedOptions& options, std::string& outName)
    {
        zip_int64_t entryIndex = -1;
        if (options.entryName)
        {
            entryIndex = zip_name_locate(archive, options.entryName->c_str(), ZIP_FL_NOCASE);
            if (entryIndex < 0)
            {
                throw std::runtime_error("Can't find entry:  " + *options.entryName);
            }
            outName+=*options.entryName;
        }
        else
        {
            auto count = zip_get_num_entries(archive, 0);
            if (count < 0)
            {
                throw std::runtime_error("Unable to get number of entries");
            }
            for (zip_uint64_t i = 0; i < static_cast<zip_uint64_t>(count); ++i)
            {
                struct zip_stat stat;
                zip_stat_init(&stat);
                if (zip_stat_index(archive, i, 0, &stat) == 0)
                {
                    std::string name = stat.name ? stat.name : "" ;
                    if (!stat.name || stat.name[0] == '\0')
                    {
                        continue;
                    }
                    if (name.size() >=4 && name.rfind(".csv") == name.size() - 4)
                    {
                        entryIndex = static_cast<zip_int64_t>(i);
                        outName = std::move(name);
                        break;
                    }
                }
            }
            if (entryIndex < 0)
            {
                throw std::runtime_error("Can't find entry in ZIP");
            }
        }
        zip_file_t* handle = zip_fopen_index(archive, entryIndex, 0);
        if (!handle)
        {
            throw std::runtime_error("Failed to open ZIP entry");
        }
        return handle;
    }

bool bulk::seedCompaniesFromZipCsvFile(const std::string& zipPath, PGconn* conn, const ZipSeedOptions& options, SeedStats* statsOut)
{
    if (!conn || PQstatus(conn) != CONNECTION_OK)
    {
        throw std::runtime_error("seedCompaniesFromZipCsvFile: invalid PG connection");
    }

    int zipError = 0;
    auto zipCloser = [](zip_t* z )
    {
        if (z)
        {
            zip_close(z);
        }
    };

    std::unique_ptr<zip_t, decltype(zipCloser)> archive(zip_open(zipPath.c_str(), ZIP_RDONLY, &zipError), zipCloser);
    if (!archive)
    {
        zip_error_t zipErr;
        zip_error_init(&zipErr);
        zip_error_set(&zipErr, zipError, errno);
        const char* errStr = zip_error_strerror(&zipErr);
        const std::string errMsg = errStr ? errStr : "unknown zip error";
        zip_error_fini(&zipErr);
        throw std::runtime_error(std::string("zip_open failed: ") + errMsg);
    }

    std::string entryName;
    auto entryCloser = [](zip_file_t* f)
    {
        if (f)
        {
            zip_fclose(f);
        }
    };
    std::unique_ptr<zip_file_t, decltype(entryCloser)> entry(openTargetEntry(archive.get(), options, entryName), entryCloser);
    SeedStats stats{};
    pg::CopySession copy(conn, buildCompaniesCopySql(options.targetTable));

    LineAccumulator accumulator;
    bool headerParsed = false;
    BasicCompanyHeader schema;

    auto processLine = [&](std::string_view line)
    {
        if (!headerParsed)
        {
            auto headerFields = parseCsvLine(line);
            schema = buildHeader(headerFields);
            if (!schema.valid())
            {
                throw std::runtime_error("Basic company CSV header missing mandatory columns");
            }
            headerParsed = true;
            return;
        }

        stats.linesTotal++;
        auto fields = parseCsvLine(line);
        if (fields.size() < 3)
        {
            stats.linesBad++;
            return;
        }

        CompanyRecord rec;
        const bool ok = populateCompanyRecord(fields, schema, rec, options.allowPartialRow);
        if (!ok)
        {
            stats.linesBad++;
            return;
        }
        const std::string csv = toCompanyCsvLine(rec);
        copy.putCsvLine(csv);
        stats.linesOk++;
    };

    std::array<char, 256* 1024> buffer{};
    while (true)
    {
        const zip_int64_t read = zip_fread(entry.get(), buffer.data(), buffer.size());
        if (read < 0)
        {
            throw std::runtime_error("Failed to read ZIP file: " + entryName);
        }
        if (read == 0)
        {
            break;
        }
        std::string_view chunk(buffer.data(), static_cast<size_t>(read));
        accumulator.feed(chunk, processLine);
    }
    accumulator.flush(processLine);
    copy.finish();

    if (statsOut)
    {
        *statsOut = stats;
    }

return true;

}

    Conn::Conn(const std::string& connInfo)
    {
        mConn = PQconnectdb(connInfo.c_str());
        if (!mConn || PQstatus(mConn) != CONNECTION_OK)
        {
            auto err = error();
            close();
            throw std::runtime_error("PQconnectDb failed: " + err );
        }
    }

    Conn::~Conn()
    {
        close();
    }

    Conn::Conn(Conn&& other) noexcept : mConn(std::move(other.mConn))
    {
        other.mConn = nullptr;
    }

    Conn& Conn::operator=(Conn&& other) noexcept
    {
        if (this != &other)
        {
            close();
            mConn = other.mConn;
            other.mConn = nullptr;
        }

        return *this;
    }

    void Conn::close()
    {
        if (mConn)
        {
            PQfinish(mConn);
            mConn = nullptr;
        }
    }

    std::string Conn::error() const
    {
        if (!mConn)
        {
            return "no connection";
        }
        auto * e = PQerrorMessage(mConn);
        return e ? std::string(e) : std::string();
    }

    Copier::Copier(PGconn* conn) : mConn(conn)
    {
        if (!mConn || PQstatus(mConn) != CONNECTION_OK)
        {
            throw std::runtime_error("Copier: invalid PGConn");
        }
    }

    void Copier::startCopy(const std::string& copySql) const
    {
        PGresult *res = PQexec(mConn, copySql.c_str());
        if (PQresultStatus(res) != PGRES_COPY_IN)
        {
            const std::string err = PQerrorMessage(mConn);
            PQclear(res);
            throw std::runtime_error("COPY Start failed: " + err);
        }

        PQclear(res);
    }

    void Copier::putLine(const std::string& line) const
    {
        if (PQputCopyData(mConn, line.c_str(), static_cast<int>(line.size())) != 1)
        {
            throw std::runtime_error("PQputCopyData failed");
        }
        if (PQputCopyData(mConn, "\n", 1) != 1)
        {
            throw std::runtime_error("PQputCopyData new line failed");
        }
    }

    void Copier::finishCopy() const
    {
        if (PQputCopyEnd(mConn, nullptr) != 1)
        {
            throw std::runtime_error("PQputCopyEnd failed");
        }
        PGresult* res = PQgetResult(mConn);
        if (PQresultStatus(res) != PGRES_COMMAND_OK)
        {
            std::string err = PQerrorMessage(mConn);
            PQclear(res);
            throw std::runtime_error("COPY Commit failed: " + err);
        }
        PQclear(res);
    }

    void Copier::copy(const std::string& copySql, const std::function<bool(std::string&)>& nextLine)
    {
        startCopy(copySql);
        std::string line;
        while (nextLine(line))
        {
            putLine(line);
        }
        finishCopy();
    }

    void Copier::copy(const std::string& copySql, const std::vector<std::string>& lines)
    {
        startCopy(copySql);
        for (const auto& line : lines)
        {
            putLine(line);
        }
        finishCopy();
    }

    struct FileCleanUpGuard
    {
        fs::path p;
        ~FileCleanUpGuard()
        {
            if (!p.empty() && fs::exists(p))
            {
                std::error_code ec;
                fs::remove(p, ec);
                if (ec)
                {
                    PLOGW << "[PscIngestService] Failed to remove file " << p.string()
                          << ": " << ec.message();
                }
                else
                {
                    PLOGD << "[PscIngestService] Removed file " << p.string();
                }
            }
        }
    };
         
    // Returns previous business day, excludes weekends.
    // If today is Monday, returns Friday. Otherwise, return previous day if it's a weekday
    std::chrono::year_month_day defaultSnapshotDate()
    {
        using namespace std::chrono;
        auto now = system_clock::now();
        auto today = floor<days>(now);

        auto dp = sys_days{year_month_day{today}} - days{1};
        weekday wd{dp};
        unsigned dayOfWeek = wd.c_encoding();

        // If yesterday was sunday, go back to Friday - 2 days
        if (dayOfWeek == 0)
        {
            dp = dp - days{2};
        }
        else if ( dayOfWeek == 6) // Saturday go back one day
        {
            dp = dp - days{1};
        }

        auto ymd = year_month_day{dp};
        return ymd;
    }

   bool PscIngestService::ingestDailySnapshot(pg_conn* conn,
                                              std::optional<std::chrono::year_month_day> dateOverride,
                                              PscSnapshotMode mode,
                                              SeedStats* out,
                                              const std::string& targetTable,
                                              bool truncateBeforeLoad,
                                              std::optional<std::string>* snapshotDateOut) const
   {
        if (!conn || PQstatus(conn) != CONNECTION_OK)
        {
            PLOGE << "PscIngestService::ingestDailySnapshot: invalid PG connection";
            return false;
        }
        if (!isSimpleSqlIdentifier(targetTable))
        {
            PLOGE << "PscIngestService::ingestDailySnapshot: invalid target table: " << targetTable;
            return false;
        }

        namespace fs = std::filesystem;
        fs::path downloadDir = mConfig.bulkDownloadDir;
        if (!downloadDir.empty() && !fs::exists(downloadDir))
        {
            std::error_code ec;
            fs::create_directories(downloadDir, ec);
            if (ec)
            {
                PLOGE << "[PSCService] Failed to create download directory "
                      << downloadDir.string() << ": " << ec.message();
                return false;
            }
        }

        SeedStats total{};
        PLOGI << "[PSCService] Mode: " << ((mode == Split30) ? "Split" : "SingleZip");
        PLOGI << "[PSCService] Download dir: " << downloadDir.string();

        auto ingestOneZip = [&](const std::string& url, const std::string& label) -> bool
        {
            const std::string filename = url.substr(url.find_last_of('/') + 1);
            fs::path zipPath = downloadDir / filename;
            FileCleanUpGuard g{zipPath};

            PLOGI << "[PSCService] " << label << " URL: " << url;
            PLOGI << "[PSCService] Downloading to: " << zipPath.string();

            const bool downloaded = mHttp.downloadFile(
                url,
                zipPath.string(),
                {},
                [](size_t now, size_t totalBytes)
                {
                    if (totalBytes > 0 && (now % (1024 * 1024 * 10) == 0))
                    {
                        PLOGD << "[PSCService] Downloaded: "
                              << (now / (1024 * 1024)) << "MB / "
                              << (totalBytes / (1024 * 1024)) << "MB";
                    }
                });

            if (!downloaded)
            {
                PLOGE << "[PSCService] Download failed: " << url;
                return false;
            }

            PLOGI << "[PSCService] Download complete: " << zipPath.string();

            PscSeedOptions seedOpt{};
            seedOpt.allowPartial = true;
            seedOpt.verbose = true;
            seedOpt.targetTable = targetTable;

            SeedStats partStats{};
            const bool ok = seedPscFromZipSnapshot(zipPath.string(), conn, seedOpt, &partStats);
            if (!ok)
            {
                PLOGE << "[PSCService] Seeding failed for: " << zipPath.string();
                return false;
            }

            accumulate(total, partStats);

            PLOGI << "[PSCService] " << label << " done. ok=" << partStats.linesOk
                  << " bad=" << partStats.linesBad
                  << " total=" << partStats.linesTotal;

            return true;
        };

        // ---- Source of truth: listing page ----
        const std::string listingUrl = "https://download.companieshouse.gov.uk/en_pscdata.html";
        std::string html;
        {
            const bool got = mHttp.getText(listingUrl, {{"Accept","text/html"}}, html);
            if (!got || html.empty())
            {
                PLOGE << "[PSCService] Failed to download listing HTML: " << listingUrl;
                return false;
            }
        }

        std::vector<PscPartLink> links = extractPscPartLinksFromHtml(html);
        if (links.empty())
        {
            PLOGE << "[PSCService] No PSC links found on listing page.";
            return false;
        }

        // ---- Decide plan (single vs split, override vs latest) ----
        std::vector<PscPartLink> parts;
        try
        {
            if (mode == SingleZip)
            {
                const std::optional<std::string> requestedDate = dateOverride
                    ? std::optional<std::string>{bulk::fmtDate(*dateOverride)}
                    : std::nullopt;

                std::optional<std::string> pageLatestDate = extractPscLatestSnapshotDateFromHtml(html);
                if (!requestedDate && !pageLatestDate)
                {
                    PLOGE << "[PSCService] Could not determine latest PSC snapshot date from listing page.";
                    return false;
                }
                const std::string dateToUse = requestedDate.value_or(*pageLatestDate);

                const auto singleHrefOpt = extractPscSingleZipHrefFromHtml(html);
                if (!singleHrefOpt)
                {
                    PLOGE << "[PSCService] No single PSC snapshot ZIP found on listing page.";
                    return false;
                }

                const std::string singleUrl = *singleHrefOpt;
                if (singleUrl.find(dateToUse) == std::string::npos)
                {
                    PLOGE << "[PSCService] Single ZIP on listing page does not match requested date "
                          << dateToUse << ". URL was: " << singleUrl;
                    return false;
                }

                PscPartLink p;
                p.url = singleUrl;
                p.date = dateToUse;
                p.part = 1;
                p.total = 1;
                parts.push_back(std::move(p));
            }
            else
            {
                if (dateOverride)
                {
                    const std::string requested = bulk::fmtDate(*dateOverride);
                    parts = filterAndSortParts(std::move(links), requested);
                    if (parts.empty())
                    {
                        PLOGE << "[PSCService] No PSC split parts found for requested date "
                              << requested << " on listing page.";
                        return false;
                    }
                    validateContiguousPartsOrThrow(parts);
                }
                else
                {
                    parts = selectLatestSnapshotPartsOrThrow(std::move(links));
                    validateContiguousPartsOrThrow(parts);
                }
            }
        }
        catch (const std::exception& e)
        {
            PLOGE << "[PSCService] Failed to build ingestion plan: " << e.what();
            return false;
        }

        if (parts.empty())
        {
            PLOGE << "[PSCService] No PSC parts selected for ingest.";
            return false;
        }

        // At this point we know ingestion can proceed.
        PLOGI << "[PSCService] Using snapshot date " << parts.front().date
              << " with " << parts.front().total << " part(s)"
              << " into table " << targetTable;
        if (snapshotDateOut)
        {
            *snapshotDateOut = parts.front().date;
        }

        if (truncateBeforeLoad)
        {
            const std::string truncateSql = "TRUNCATE TABLE " + targetTable;
            PGresult* res = PQexec(conn, truncateSql.c_str());
            if (!res || PQresultStatus(res) != PGRES_COMMAND_OK)
            {
                std::string err = PQerrorMessage(conn);
                if (res) PQclear(res);
                PLOGE << "[PSCService] TRUNCATE " << targetTable << " failed: " << err;
                return false;
            }
            PQclear(res);
        }

        //  Ingest
        for (const auto& p : parts)
        {
            const std::string label = fmt::format("[{}/{}]", p.part, p.total);
            if (!ingestOneZip(p.url, label))
            {
                return false;
            }
        }

        PLOGI << "[PSCService] Done. Total lines=" << total.linesTotal
              << " ok=" << total.linesOk
              << " bad=" << total.linesBad;

        if (out) *out = total;
        return true;
    }

    bool populatePscRecord(std::string_view line, PscRecord& out, bool allowPartial)
    {
        auto j = nlohmann::json::parse(line, nullptr, false);
        if (j.is_discarded())
        {
            return false;
        }

        if (!j.contains("company_number") || !j.contains("data"))
        {
            return false;
        }

        const auto& data = j.at("data");
        if (!data.is_object())
        {
            return false;
        }

        if (!j["company_number"].is_string())
        {
            return false;
        }

        out.companyNumber = j["company_number"].get<std::string>();
        out.kind = data.value("kind", "");
        out.name.reset();
        if (data.contains("name") && data["name"].is_string())
        {
            out.name = data["name"].get<std::string>();
        }

        out.pscKey.clear();

        if (data.contains("links") && data["links"].is_object())
        {
            const auto& links = data["links"];
            if (links.contains("self") && links["self"].is_string())
            {
                out.pscKey = links["self"].get<std::string>();
                out.linkSelf = out.pscKey;
            }
        }

        // Use this as fallback
        if (out.pscKey.empty())
        {
            out.pscKey = out.companyNumber  + "|" + out.name.value();
        }

        out.countryOfResidence = data.value("country_of_residence", "");
        out.nationality        = data.value("nationality", "");

        out.dobYear.reset();
        out.dobMonth.reset();
        if (data.contains("date_of_birth") && data["date_of_birth"].is_object())
        {
            const auto& dob = data["date_of_birth"];
            if (dob.contains("year") && dob["year"].is_number_integer())
            {
                out.dobYear = dob["year"].get<int>();
            }
            if (dob.contains("month") && dob["month"].is_number_integer())
            {
                out.dobMonth = dob["month"].get<int>();
            }
        }

        out.notifiedOn = data.value("notified_on", "");
        out.ceasedOn   = data.value("ceased_on", "");

        out.naturesOfControl.clear();
        if (data.contains("natures_of_control") && data["natures_of_control"].is_array())
        {
            for (const auto& n : data["natures_of_control"])
            {
                if (n.is_string())
                    out.naturesOfControl.push_back(n.get<std::string>());
            }
        }

        out.rawJson.assign(line);

        return true;

    bool seedPscFromZipSnapshotIntoSink(const std::string& zipPath, ICsvCopySink& sink, const PscSeedOptions& options, SeedStats* statsOut)
    {
        SeedStats stats{};

        int zipError = 0;
        auto zipCloser = [](zip_t* z){ if (z) zip_close(z); };

        std::unique_ptr<zip_t, decltype(zipCloser)> archive(zip_open(zipPath.c_str(), ZIP_RDONLY, &zipError),zipCloser);
        if (!archive)
        {
            zip_error_t zipErr;
            zip_error_init(&zipErr);
            zip_error_set(&zipErr, zipError, errno);
            const char* errStr = zip_error_strerror(&zipErr);
            const std::string errMsg = errStr ? errStr : "unknown zip error";
            zip_error_fini(&zipErr);
            throw std::runtime_error("Failed to open zip file: " + errMsg);
        }

        zip_int64_t entryIndex = -1;
        std::string entryName;

        const zip_uint64_t count = zip_get_num_entries(archive.get(), 0);
        for (zip_uint64_t i = 0; i < count; ++i)
        {
            zip_stat_t st{};
            zip_stat_init(&st);

            if (zip_stat_index(archive.get(), i, 0, &st) == 0 && st.name)
            {
                std::string name = st.name;
                if (name.size() >= 4)
                {
                    std::string ext = name.substr(name.size() - 4);
                    std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);

                    if (ext == ".txt")
                    {
                        entryIndex = static_cast<zip_int64_t>(i);
                        entryName  = std::move(name);
                        break;
                    }
                }
            }
        }

        if (entryIndex < 0)
        {
            throw std::runtime_error("Failed to find .txt entry in zip file: " + zipPath);
        }

        auto fileCloser = [](zip_file_t* f){ if (f) zip_fclose(f); };

        std::unique_ptr<zip_file_t, decltype(fileCloser)> entry(zip_fopen_index(archive.get(), entryIndex, 0),fileCloser);

        if (!entry)
        {
            throw std::runtime_error("Failed to open entry in zip file: " + entryName);
        }

        LineAccumulator accumulator;
        auto processLine = [&](std::string_view sv)
        {
            if (sv.empty())
            {
                return;
            }
            auto line = trim(sv);
            if (line.empty())
            {
                return;
            }
            stats.linesTotal++;
            PscRecord record;
            if (!populatePscRecord(line, record, options.allowPartial))
            {
                stats.linesBad++;
                if (!options.allowPartial)
                    throw std::runtime_error("Failed to parse PSC record (bad JSON/object)");
                return;
            }

            sink.putCsvLine(toPscCsvLine(record));
            stats.linesOk++;

            if (options.verbose && (stats.linesOk % 1000000 == 0))
            {
                PLOGI << "[PSCSeed] " << stats.linesOk << " ok, " << stats.linesBad << " bad";
            }
        };

        std::array<char, 256 * 1024> buffer{};
        while (true)
        {
            const zip_int64_t n = zip_fread(entry.get(), buffer.data(), buffer.size());
            if (n < 0)
                throw std::runtime_error("Failed to read entry in zip file: " + entryName);

            if (n == 0) break;

            accumulator.feed(std::string_view(buffer.data(), static_cast<size_t>(n)), processLine);
        }

        accumulator.flush(processLine);
        sink.finish();

        if (statsOut)
        {
            *statsOut = stats;
        }
        return true;
    }

    std::string normaliseDownloadUrl(std::string href)
    {
        if (href.rfind("http://", 0) == 0 || href.rfind("https://", 0) == 0)
        {
            return href;
        }

        if (!href.empty() && href.front() == '/')
        {
            return "https://download.companieshouse.gov.uk" + href;
        }

        return "https://download.companieshouse.gov.uk/" + href;
    }

    static inline std::string filenameFromUrl(const std::string& url)
    {
        const auto slash = url.rfind("/");
        if (slash == std::string::npos)
        {
            return url;
        }
        return url.substr(slash + 1);
    }


    std::vector<AccountsZipLink> extractDailyAccountsLinksFromHtml(const std::string& html)
    {
        // Daily links look like:
        //   href="https://download.companieshouse.gov.uk/Accounts_Bulk_Data-2025-12-30.zip"
        // Sometimes size appears in the anchor text: "...zip  (255Mb)"
        // Group 1: href value
        // Group 2: date YYYY-MM-DD
        // Group 3: size (optional) captured without parentheses, e.g. "255Mb"
        static const std::regex re(
           R"delim(href="([^"]*Accounts_Bulk_Data-(\d{4}-\d{2}-\d{2})\.zip)"[^<]*</a>\s*(?:\(\s*([^)]*?)\s*\))?)delim",
           std::regex::icase);

        std::vector<AccountsZipLink>out;
        out.reserve(64);

        for (std::sregex_iterator it(html.begin(), html.end(), re), end; it != end; ++it)
        {
            AccountsZipLink link{};
            link.source = AccountsSource::Daily;
            link.url = normaliseDownloadUrl((*it)[1].str());
            link.key = (*it)[2].str();
            link.filename  = filenameFromUrl(link.url);
            if ((*it)[3].matched)
            {
                link.sizeText = (*it)[3].str();
            }
            out.push_back(std::move(link));
        }
        return out;
    }
    std::vector<AccountsZipLink> extractMonthlyAccountsLinksFromHtml(const std::string& html, AccountsSource source)
    {
        // Monthly/Historic links look like:
        //   Accounts_Monthly_Data-November2025.zip
        // Historic also includes:
        //   Accounts_Monthly_Data-JanuaryToDecember2008.zip
        //
        // Group 1: href value
        // Group 2: key after dash (MonthNameYYYY or JanuaryToDecemberYYYY)
        // Group 3: size (optional)
        static const std::regex re(

std::string gz::decompressBuffer(const void* data, size_t len, HeaderMode mode)
{
    Inflater inflater(mode);
    std::string out;
    out.reserve(len*2);
    size_t offset =0;
    auto push = [&](const uint8_t* p, size_t n)
    {
        out.append(reinterpret_cast<const char*>(p), n);
    };

    while (offset < len)
    {
        size_t chunk = std::min<size_t>(64 * 1024, len - offset);
        inflater.feed(reinterpret_cast<const uint8_t*>(data) + offset, chunk, push);
        offset += chunk;
    }
    inflater.finish(push);
    return out;
}

bool gz::decompressFileToFile(const std::string& inPath, const std::string& outPath, HeaderMode mode, size_t chunk)
{
    std::ifstream ifs(inPath, std::ios::binary);
    if (!ifs)
    {
        return false;
    }
    std::ofstream ofs(outPath, std::ios::binary | std::ios::trunc);
    if (!ofs)
    {
        return false;
    }

    Inflater inflater(mode);
    std::vector<uint8_t> buffer(chunk);
    auto push = [&](const uint8_t* p, size_t n)
    {
       ofs.write(reinterpret_cast<const char*>(p), n);
    };

    while (ifs)
    {
        ifs.read(reinterpret_cast<char*>(buffer.data()), static_cast<std::streamsize>(buffer.size()));
        std::streamsize got = ifs.gcount();
        if (got > 0)
        {
            inflater.feed(reinterpret_cast<const uint8_t*>(buffer.data()), static_cast<size_t>(got), push);
        }
    }

    inflater.finish(push);
    ofs.flush();
    return ofs.good();

}

std::chrono::hours parseHoursNode(const YAML::Node& node,
                                  std::chrono::hours defaultValue)
{
    if (!node || !node.IsScalar())
        return defaultValue;

    const auto s = node.as<std::string>();

    // digits-only string: "24"
    if (!s.empty() && std::ranges::all_of(s.begin(), s.end(),
        [](const unsigned char c) { return std::isdigit(c) != 0; }))
    {
        try
        {
            return std::chrono::hours{std::stoull(s)};
        }
        catch (...)
        {
            return defaultValue;
        }
    }

    // "36h"
    if (!s.empty() && s.back() == 'h')
    {
        try
        {
            return std::chrono::hours{std::stoull(s.substr(0, s.size() - 1))};
        } catch (...)
        {
            return defaultValue;
        }
    }

    try
        {
        return std::chrono::hours{node.as<long long>()};
    }
    catch (...) {
    }

    return defaultValue;
}

std::chrono::milliseconds parseMillisNode(const YAML::Node& node, std::chrono::milliseconds defaultValue)
{
    if (!node || !node.IsScalar())
    {
        return defaultValue;
    }

    try
    {
        return std::chrono::milliseconds{node.as<long long>()};
    }
    catch (...)
    {
        return defaultValue;
    }
}

std::chrono::milliseconds parseDurationNode(const YAML::Node& node,
                                            std::chrono::milliseconds defaultValue)
{
    if (!node || !node.IsScalar())
    {
        return defaultValue;
    }

    const auto parseNumeric = [](const std::string& s) -> std::optional<long long>
    {
        if (s.empty() || !std::ranges::all_of(s.begin(), s.end(),
            [](const unsigned char c) { return std::isdigit(c) != 0; }))
        {
            return std::nullopt;
        }
        try
        {
            return std::stoll(s);
        }
        catch (...)
        {
            return std::nullopt;
        }
    };

    auto raw = node.as<std::string>();
    raw.erase(std::remove_if(raw.begin(), raw.end(),
        [](const unsigned char c) { return std::isspace(c) != 0; }),
        raw.end());
    if (raw.empty())
    {
        return defaultValue;
    }

    std::string lowered = raw;
    std::transform(lowered.begin(), lowered.end(), lowered.begin(),
                   [](const unsigned char c) { return static_cast<char>(std::tolower(c)); });

    auto millisFromCount = [](long long count, long long factor) -> std::optional<std::chrono::milliseconds>
    {
        if (count <= 0 || factor <= 0)
        {
            return std::nullopt;
        }
        return std::chrono::milliseconds{count * factor};
    };

    if (const auto v = parseNumeric(lowered))
    {
        if (auto ms = millisFromCount(*v, 1))
        {
            return *ms;
        }
        return defaultValue;
    }

    auto parseSuffix = [&](const char* suffix, long long factor) -> std::optional<std::chrono::milliseconds>
    {
        const std::string suffixStr = suffix;
        if (lowered.size() <= suffixStr.size() ||
            lowered.rfind(suffixStr) != lowered.size() - suffixStr.size())
        {
            return std::nullopt;
        }
        const std::string number = lowered.substr(0, lowered.size() - suffixStr.size());
        const auto v = parseNumeric(number);
        if (!v)
        {
            return std::nullopt;
        }
        return millisFromCount(*v, factor);
    };

    if (auto ms = parseSuffix("ms", 1)) return *ms;
    if (auto ms = parseSuffix("s", 1000)) return *ms;
    if (auto ms = parseSuffix("m", 60 * 1000)) return *ms;
    if (auto ms = parseSuffix("h", 60 * 60 * 1000)) return *ms;
    if (auto ms = parseSuffix("d", 24 * 60 * 60 * 1000)) return *ms;

    return defaultValue;
}

std::size_t parseSizeT(const YAML::Node& node, std::size_t defaultValue)
{
    if (!node)
    {
        return defaultValue;
    }

    try
    {
        auto n = node.as<long long>();
        if (n <= 0)
        {
            return defaultValue;
        }
        return n;
    } catch (...)
    {
        return defaultValue;
    }
}

int parsePositiveInt(const YAML::Node& node, int defaultValue)
{
    if (!node)
    {
        return defaultValue;
    }
    try
    {
        const auto n = node.as<int>();
        return n > 0 ? n : defaultValue;
    }
    catch (...)
    {
        return defaultValue;
    }
}

    std::string trim(std::string_view sv)
    {
        const auto* begin = sv.data();
        const auto* end = begin + sv.size();

        while (begin < end && std::isspace(static_cast<unsigned char>(*begin)))
        {
            ++begin;
        }

        while (end > begin && std::isspace(static_cast<unsigned char>(*(end - 1))))
        {
            --end;
        }

        return {begin, end};
    }

    std::vector<std::string> parseCsvLine(std::string_view line)
    {
        std::vector<std::string> fields;
        std::string current;
        current.reserve(line.size());

        bool inQuotes = false;
        for (size_t i = 0; i < line.size(); ++i)
        {
            char ch = line[i];
            if (inQuotes)
            {
                if (ch == '"')
                {
                    if (i + 1 < line.size() && line[i + 1] == '"')
                    {
                        current.push_back('"');
                        ++i;
                    }
                    else
                    {
                        inQuotes = false;
                    }
                }
                else
                {
                    current.push_back(ch);
                }
            }
            else
            {
                if (ch == '"')
                {
                    inQuotes = true;
                }
                else if (ch == ',')
                {
                    fields.emplace_back(std::move(current));
                    current.clear();
                }
                else if (ch == '\r')
                {
                    // ignore the CR
                }
                else
                {
                    current.push_back(ch);
                }
            }
        }
        fields.emplace_back(std::move(current));
        return fields;
    }

    bool isSimpleSqlIdentifier(const std::string_view value)
    {
        if (value.empty())
        {
            return false;
        }

        const auto isAsciiAlpha = [](const unsigned char c)
        {
            return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z');
        };
        const auto isAsciiDigit = [](const unsigned char c)
        {
            return c >= '0' && c <= '9';
        };

        const unsigned char first = static_cast<unsigned char>(value.front());
        if (!(isAsciiAlpha(first) || first == '_'))
        {
            return false;
        }

        for (const char ch : value)
        {
            const unsigned char c = static_cast<unsigned char>(ch);
            if (!(isAsciiAlpha(c) || isAsciiDigit(c) || c == '_'))
            {
                return false;
            }
        }

        return true;
    }

    std::optional<std::string> normalizeIxbrlNumeric(std::string_view valueText,
                                                     std::string_view sign,
                                                     std::string_view scale)
    {
        std::string cleaned;
        cleaned.reserve(valueText.size());
        for (char c : valueText)
        {
            if (c == ',' || std::isspace(static_cast<unsigned char>(c)) != 0)
            {
                continue;
            }
            if (std::isdigit(static_cast<unsigned char>(c)) != 0 || c == '.' || c == '-' || c == '+')
            {
                cleaned.push_back(c);
            }
            else
            {
                return std::nullopt;
            }
        }

        if (cleaned.empty())
        {
            return std::nullopt;
        }

        bool negative = !sign.empty() && sign[0] == '-';
        if (cleaned[0] == '-')
        {
            negative = true;
            cleaned.erase(cleaned.begin());
        }
        else if (cleaned[0] == '+')
        {
            cleaned.erase(cleaned.begin());
        }

        if (cleaned.empty())
        {
            return std::nullopt;
        }

        bool seenDot = false;
        for (char c : cleaned)
        {
            if (c == '.')
            {
                if (seenDot)
                {
                    return std::nullopt;
                }
                seenDot = true;
                continue;
            }
            if (std::isdigit(static_cast<unsigned char>(c)) == 0)
            {
                return std::nullopt;
            }
        }

        std::string normalized = cleaned;
        if (auto scaleValue = parseScale(scale))
        {
            const auto pos = normalized.find('.');
            std::string intPart = pos == std::string::npos ? normalized : normalized.substr(0, pos);
            std::string fracPart = pos == std::string::npos ? "" : normalized.substr(pos + 1);
            std::string allDigits = intPart + fracPart;
            if (allDigits.empty())
            {
                return std::nullopt;
            }

            int decimalPos = static_cast<int>(intPart.size());
            int targetPos = decimalPos + *scaleValue;

            if (targetPos <= 0)
            {
                normalized = "0." + std::string(static_cast<std::size_t>(-targetPos), '0') + allDigits;
            }
            else if (targetPos >= static_cast<int>(allDigits.size()))
            {
                normalized = allDigits + std::string(static_cast<std::size_t>(targetPos - static_cast<int>(allDigits.size())), '0');
            }
            else
            {
                normalized = allDigits.substr(0, static_cast<std::size_t>(targetPos)) + "." +
                             allDigits.substr(static_cast<std::size_t>(targetPos));
            }
        }

        if (negative && !normalized.empty() && normalized[0] != '-')
        {
            normalized.insert(normalized.begin(), '-');
        }

        return normalized;
    }

        std::string toLower(std::string_view s)
        {
            std::string out;
            out.reserve(s.size());
            for (char c : s)
            {
                out.push_back(static_cast<char>(std::tolower(static_cast<unsigned char>(c))));
            }
            return out;
        }

        bool hasExtension(const std::string& name, const char* ext)
        {
            const std::string lower = toLower(name);
            const std::string extLower = toLower(ext);
            if (lower.size() < extLower.size())
            {
                return false;
            }
            return lower.rfind(extLower) == lower.size() - extLower.size();
        }

        bool isIxbrlEntry(const std::string& name)
        {
            return hasExtension(name, ".xhtml") || hasExtension(name, ".html") || hasExtension(name, ".ixbrl");
        }

        bool isXbrlEntry(const std::string& name)
        {
            return hasExtension(name, ".xml") || hasExtension(name, ".xbrl");
        }

        std::string readZipEntry(zip_t* archive, zip_uint64_t index, std::size_t maxBytes)
        {
            zip_stat_t st;
            zip_stat_init(&st);
            if (zip_stat_index(archive, index, 0, &st) != 0)
            {
                throw std::runtime_error("Failed to stat ZIP entry");
            }

            if (maxBytes > 0 && st.size > static_cast<zip_uint64_t>(maxBytes))
            {
                throw std::runtime_error("ZIP entry exceeds maxBytes");
            }

            zip_file_t* handle = zip_fopen_index(archive, index, 0);
            if (!handle)
            {
                throw std::runtime_error("Failed to open ZIP entry");
            }

            std::string out;
            out.reserve(static_cast<std::size_t>(st.size));
            std::array<char, 256 * 1024> buffer{};
            while (true)
            {
                zip_int64_t read = zip_fread(handle, buffer.data(), buffer.size());
                if (read < 0)
                {
                    zip_fclose(handle);
                    throw std::runtime_error("Failed to read ZIP entry");
                }
                if (read == 0)
                {
                    break;
                }
                out.append(buffer.data(), static_cast<std::size_t>(read));
                if (maxBytes > 0 && out.size() > maxBytes)
                {
                    zip_fclose(handle);
                    throw std::runtime_error("ZIP entry exceeds maxBytes while reading");
                }
            }
            zip_fclose(handle);
            return out;
        }

        const char* docTypeToString(ixbrl::DocumentMeta::Type type)
        {
            switch (type)
            {
                case ixbrl::DocumentMeta::Type::IXBRL:
                    return "IXBRL";
                case ixbrl::DocumentMeta::Type::XBRL:
                    return "XBRL";
            }
            return "IXBRL";
        }

        std::optional<std::string> extractCompanyNumberFromPath(const std::string& path)
        {
            static const std::regex pattern("([A-Z]{2}\\d{6}|\\d{8})");
            std::string upper;
            upper.reserve(path.size());
            for (char c : path)
            {
                upper.push_back(static_cast<char>(std::toupper(static_cast<unsigned char>(c))));
            }
            std::smatch match;
            if (std::regex_search(upper, match, pattern) && match.size() > 1)
            {
                return match[1].str();
            }
            return std::nullopt;
        }

        std::string csvEscape(std::string_view value)
        {
            bool needsQuotes = false;
            for (char c : value)
            {
                if (c == ',' || c == '"' || c == '\n' || c == '\r')
                {
                    needsQuotes = true;
                    break;
                }
            }

            if (!needsQuotes)
            {
                return std::string(value);
            }

            std::string out;
            out.reserve(value.size() + 2);
            out.push_back('"');
            for (char c : value)
            {
                if (c == '"')
                {
                    out.push_back('"');
                    out.push_back('"');
                }
                else
                {
                    out.push_back(c);
                }
            }
            out.push_back('"');
            return out;
        }

        std::string csvField(const std::optional<std::string>& value)
        {
            if (!value.has_value())
            {
                return "\\N";
            }
            return csvEscape(*value);
        }
    }

    bool AccountsIxbrlIngestService::ingestZipFile(const std::filesystem::path& zipPath, pqxx::connection& db, PGconn* copyConn,
                                                   const AccountsIxbrlIngestOptions& options,
                                                   IxbrlIngestStats* statsOut) const
    {
        int zipError = 0;
        auto zipCloser = [](zip_t* z)
        {
            if (z)
            {
                zip_close(z);
            }
        };

        std::unique_ptr<zip_t, decltype(zipCloser)> archive(
            zip_open(zipPath.string().c_str(), ZIP_RDONLY, &zipError), zipCloser);
        if (!archive)
        {
            zip_error_t zipErr;
            zip_error_init(&zipErr);
            zip_error_set(&zipErr, zipError, errno);
            const char* errStr = zip_error_strerror(&zipErr);
            const std::string errMsg = errStr ? errStr : "unknown zip error";
            zip_error_fini(&zipErr);
            throw std::runtime_error(std::string("zip_open failed: ") + errMsg);
        }

        IxbrlIngestStats stats;
        const zip_int64_t count = zip_get_num_entries(archive.get(), 0);
        if (count < 0)
        {
            throw std::runtime_error("Unable to get ZIP entry count");
        }
        const std::size_t entryCount = static_cast<std::size_t>(count);
        const std::size_t progressEvery = [&]()
        {
            if (entryCount <= 100)
            {
                return std::size_t{10};
            }
            if (entryCount <= 500)
            {
                return std::size_t{50};
            }
            if (entryCount <= 2000)
            {
                return std::size_t{100};
            }
            if (entryCount <= 10000)
            {
                return std::size_t{250};
            }
            return std::size_t{500};
        }();
        auto maybeLogProgress = [&](const IxbrlIngestStats& stats)
        {
            if (!options.verbose || progressEvery == 0)
            {
                return;
            }
            if (stats.documentsTotal > 0 && (stats.documentsTotal % progressEvery == 0))
            {
                PLOGI << "[AccountsIxbrlIngestService] Processed " << stats.documentsTotal
                      << " documents (ok=" << stats.documentsOk
                      << ", bad=" << stats.documentsBad
                      << ", facts=" << stats.factsInserted << ")";
            }
        };

        if (options.verbose)
        {
            PLOGI << "[AccountsIxbrlIngestService] Starting ingest for " << zipPath.string()
                  << " with " << entryCount << " ZIP entries";
        }

        for (zip_uint64_t i = 0; i < static_cast<zip_uint64_t>(count); ++i)
        {
            zip_stat_t st;
            zip_stat_init(&st);
            if (zip_stat_index(archive.get(), i, 0, &st) != 0)
            {
                continue;
            }
            if (!st.name || st.name[0] == '\0')
            {
                continue;
            }
            std::string entryName = st.name;
            if (!entryName.empty() && entryName.back() == '/')
            {
                continue;
            }
            if (!isIxbrlEntry(entryName) && !isXbrlEntry(entryName))
            {
                continue;
            }

            stats.documentsTotal++;
            try
            {
                std::string bytes = readZipEntry(archive.get(), i, options.parseOptions.maxBytes);
                ixbrl::Expected<ixbrl::Document> docExp = isXbrlEntry(entryName)
                    ? ixbrl::parseXbrl(bytes, options.parseOptions)
                    : ixbrl::parseIxbrl(bytes, options.parseOptions);

                if (!docExp.has_value())
                {
                    if (options.verbose)
                    {
                        PLOGW << "iXBRL parse failed for " << entryName << ": "
                              << docExp.error().message;
                    }
                    stats.documentsBad++;
                    maybeLogProgress(stats);
                    continue;
                }

                const ixbrl::Document& doc = docExp.value();
                std::optional<std::string> companyNumber = extractCompanyNumberFromFacts(doc);
                if (!companyNumber)
                {
                    companyNumber = extractCompanyNumberFromPath(entryName);
                }

                pqxx::work w{db};
                pqxx::result docInsertResult = w.exec(
                    "INSERT INTO ixbrl_documents (company_number, source_zip, source_path, doc_type) "
                    "SELECT $1, $2, $3, $4 "
                    "WHERE NOT EXISTS ("
                    "  SELECT 1 FROM ixbrl_documents WHERE source_zip=$2 AND source_path=$3"
                    ") "
                    "RETURNING id",
                    pqxx::params{companyNumber, zipPath.string(), entryName, docTypeToString(doc.meta.type)}
                );
                if (docInsertResult.empty())
                {
                    w.commit();
                    stats.documentsSkipped++;
                    maybeLogProgress(stats);
                    continue;
                }

                pqxx::row docRow = docInsertResult.one_row();
                const long long documentId = docRow[0].as<long long>();

                for (const auto& schemaRef : doc.meta.schemaRef)
                {
                    w.exec("INSERT INTO ixbrl_schema_refs (document_id, schema_ref) VALUES ($1, $2)",
                           pqxx::params{documentId, schemaRef}).no_rows();
                }

                std::unordered_map<std::string, long long> contextIds;
                contextIds.reserve(doc.contexts.size());
                for (const auto& ctx : doc.contexts)
                {
                    std::optional<std::string> instant = ctx.period.instant.empty() ? std::nullopt : std::optional<std::string>(ctx.period.instant);
                    std::optional<std::string> start = ctx.period.startDate.empty() ? std::nullopt : std::optional<std::string>(ctx.period.startDate);
                    std::optional<std::string> end = ctx.period.endDate.empty() ? std::nullopt : std::optional<std::string>(ctx.period.endDate);

                    pqxx::row ctxRow = w.exec(
                        "INSERT INTO ixbrl_contexts (document_id, context_id, entity_scheme, entity_identifier, period_kind, period_instant, period_start, period_end) "
                        "VALUES ($1, $2, $3, $4, $5, $6::date, $7::date, $8::date) RETURNING id",
                        pqxx::params{
                            documentId,
                            ctx.id,
                            ctx.entity.scheme,
                            ctx.entity.value,
                            ctx.period.kind == ixbrl::Period::Kind::Instant ? "Instant" : "Duration",
                            instant,
                            start,
                            end
                        }
                    ).one_row();
                    const long long ctxId = ctxRow[0].as<long long>();
                    contextIds.emplace(ctx.id, ctxId);
                    stats.contextsInserted++;

                    for (const auto& dm : ctx.dimensions)
                    {
                        w.exec(
                            "INSERT INTO ixbrl_context_dimensions (context_pk, dimension_raw, dimension_expanded, dimension_prefix, dimension_local, member_raw, member_expanded, member_prefix, member_local) "
                            "VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)",
                            pqxx::params{
                                ctxId,
                                dm.dimension.raw,
                                dm.dimension.expanded,
                                dm.dimension.prefix,
                                dm.dimension.local,
                                dm.member.raw,
                                dm.member.expanded,
                                dm.member.prefix,
                                dm.member.local
                            }).no_rows();
                    }
                }

                for (const auto& unit : doc.units)
                {
                    pqxx::row unitRow = w.exec(
                        "INSERT INTO ixbrl_units (document_id, unit_id) VALUES ($1, $2) RETURNING id",
                        pqxx::params{documentId, unit.id}
                    ).one_row();
                    const long long unitId = unitRow[0].as<long long>();
                    stats.unitsInserted++;

                    for (const auto& measure : unit.measures)
                    {
                        w.exec(
                            "INSERT INTO ixbrl_unit_measures (unit_pk, measure_raw, measure_expanded, measure_prefix, measure_local) "
                            "VALUES ($1, $2, $3, $4, $5)",
                            pqxx::params{unitId, measure.raw, measure.expanded, measure.prefix, measure.local}
                        ).no_rows();
                    }
                }

                w.commit();

                const std::string copySql =
                    "COPY ixbrl_facts("
                    "document_id,fact_kind,name_raw,name_expanded,name_prefix,name_local,"
                    "context_ref,unit_ref,scale,decimals,precision,format,fact_id,sign,value_text,numeric_value"
                    ") FROM STDIN WITH (FORMAT csv, NULL '\\N')";

                std::size_t factsInserted = 0;
                try
                {
                    pg::CopySession copy(copyConn, copySql);
                    for (const auto& fact : doc.facts)
                    {
                        std::optional<std::string> numericValue;
                        if (fact.kind == ixbrl::FactKind::NonFraction)
                        {
                            numericValue = normalizeIxbrlNumeric(fact.valueText, fact.sign, fact.scale);
                        }

                        std::optional<std::string> contextRef = fact.contextRef.empty() ? std::nullopt : std::optional<std::string>(fact.contextRef);
                        std::optional<std::string> unitRef = fact.unitRef.empty() ? std::nullopt : std::optional<std::string>(fact.unitRef);
                        std::optional<std::string> scale = fact.scale.empty() ? std::nullopt : std::optional<std::string>(fact.scale);
                        std::optional<std::string> decimals = fact.decimals.empty() ? std::nullopt : std::optional<std::string>(fact.decimals);
                        std::optional<std::string> precision = fact.precision.empty() ? std::nullopt : std::optional<std::string>(fact.precision);
                        std::optional<std::string> format = fact.format.empty() ? std::nullopt : std::optional<std::string>(fact.format);
                        std::optional<std::string> factId = fact.id.empty() ? std::nullopt : std::optional<std::string>(fact.id);
                        std::optional<std::string> sign = fact.sign.empty() ? std::nullopt : std::optional<std::string>(fact.sign);
                        std::optional<std::string> valueText = fact.valueText.empty() ? std::nullopt : std::optional<std::string>(fact.valueText);

                        std::string line;
                        line.reserve(256);
                        line.append(std::to_string(documentId));
                        line.push_back(',');
                        line.append(csvEscape(fact.kind == ixbrl::FactKind::NonNumeric
                                                  ? "NonNumeric"
                                                  : (fact.kind == ixbrl::FactKind::XmlFact ? "XmlFact" : "NonFraction")));
                        line.push_back(',');
                        line.append(csvEscape(fact.name.raw));
                        line.push_back(',');
                        line.append(csvField(fact.name.expanded.empty() ? std::nullopt : std::optional<std::string>(fact.name.expanded)));
                        line.push_back(',');
                        line.append(csvField(fact.name.prefix.empty() ? std::nullopt : std::optional<std::string>(fact.name.prefix)));
                        line.push_back(',');
                        line.append(csvField(fact.name.local.empty() ? std::nullopt : std::optional<std::string>(fact.name.local)));
                        line.push_back(',');
                        line.append(csvField(contextRef));
                        line.push_back(',');
                        line.append(csvField(unitRef));
                        line.push_back(',');
                        line.append(csvField(scale));
                        line.push_back(',');
                        line.append(csvField(decimals));
                        line.push_back(',');
                        line.append(csvField(precision));
                        line.push_back(',');
                        line.append(csvField(format));
                        line.push_back(',');
                        line.append(csvField(factId));
                        line.push_back(',');
                        line.append(csvField(sign));
                        line.push_back(',');
                        line.append(csvField(valueText));
                        line.push_back(',');
                        line.append(csvField(numericValue));

                        copy.putCsvLine(line);
                        factsInserted++;
                    }
                    copy.finish();
                }
                catch (...)
                {
                    pqxx::work cleanup{db};
                    cleanup.exec("DELETE FROM ixbrl_documents WHERE id=$1", pqxx::params{documentId}).no_rows();
                    cleanup.commit();
                    throw;
                }

                stats.factsInserted += factsInserted;
                stats.documentsOk++;
                maybeLogProgress(stats);
            }
            catch (const std::exception& e)
            {
                if (options.verbose)
                {
                    PLOGW << "Failed to ingest entry " << entryName << ": " << e.what();
                }
                stats.documentsBad++;
                maybeLogProgress(stats);
            }
        }

        if (statsOut)
        {
            *statsOut = stats;
        }

        if (options.verbose)
        {
            PLOGI << "[AccountsIxbrlIngestService] Finished ingest: docs=" << stats.documentsTotal
                  << " ok=" << stats.documentsOk
                  << " skipped=" << stats.documentsSkipped
                  << " bad=" << stats.documentsBad
                  << " facts=" << stats.factsInserted
                  << " contexts=" << stats.contextsInserted
                  << " units=" << stats.unitsInserted;
        }

        return stats.documentsBad == 0;
    }
}

    std::string fmtDate(std::chrono::year_month_day d)
    {
        return fmt::format("{:04}-{:02}-{:02}", static_cast<int>(d.year()), static_cast<unsigned int>(d.month()), static_cast<unsigned int>(d.day()));
    }

    std::string normalizeDownloadUrl(std::string href)
    {
        // if this is absolute just return it
        if (href.rfind("http://", 0) == 0 || href.rfind("https://", 0) == 0)
        {
            return href;
        }

        // Handle accidental leading slash handle: "/psc-snapshot-....zip"
        if (!href.empty() && href.front() == '/')
        {
            return "https://download.companieshouse.gov.uk" + href;
        }

        return "https://download.companieshouse.gov.uk/" + href;

    }

    std::vector<PscPartLink> extractPscPartLinksFromHtml(const std::string& html)
    {
        // Used to match both absolute and relative links
        // Matches BOTH:
        //   href="psc-snapshot-2025-12-29_1of31.zip"
        //   href="https://download.companieshouse.gov.uk/psc-snapshot-2025-12-29_1of31.zip"
        //
        // Group 1: full href value (relative or absolute)
        // Group 2: date
        // Group 3: part
        // Group 4: total

        static const std::regex re(
             R"delim(href="((?:https?://download\.companieshouse\.gov\.uk/)?psc-snapshot-(\d{4}-\d{2}-\d{2})_(\d+)of(\d+)\.zip)")delim",
             std::regex::icase);

        std::vector<PscPartLink> out;
        out.reserve(64);

        for (std::sregex_iterator it(html.begin(), html.end(), re), end; it!= end; ++it)
        {
            PscPartLink link;
            link.url   = normalizeDownloadUrl((*it)[1].str());
            link.date  = (*it)[2].str();
            link.part  = std::stoi((*it)[3].str());
            link.total = std::stoi((*it)[4].str());
            out.push_back(std::move(link));
        }

        return out;
    }

    std::vector<PscPartLink> filterAndSortParts(std::vector<PscPartLink> links, const std::optional<std::string>& dateFilter)
    {
        if (dateFilter)
        {
            std::erase_if(links, [&](const PscPartLink& link)
            {
                return link.date != dateFilter;
            });
        }

        std::sort(links.begin(), links.end(), [](const PscPartLink& lhs, const PscPartLink& rhs)
        {
           if (lhs.date != rhs.date)
           {
               return lhs.date < rhs.date;
           }
            return lhs.part  < rhs.part;
        });

        return links;

    }

    void validateContiguousPartsOrThrow(const std::vector<PscPartLink>& parts)
    {
        if (parts.empty())
        {
            return;
        }

        const auto date = parts.front().date;
        const int total = parts.front().total;

        for (const auto& p: parts)
        {
            if (p.date != date)
            {
                throw std::runtime_error("PSC parts validation: mixed dates in listing");
            }

            if (p.total != total)
            {
                throw std::runtime_error("PSC parts validation: mixed totals in listing");
            }
        }

        std::vector<bool> seen(total+1, false);
        for (const auto&p : parts)
        {
            if (p.part < 1 || p.part > total)
            {
                throw std::runtime_error("PSC parts validation: part index out of range");
            }
            seen[p.part] = true;
        }

        for (int i  = 1; i <= total; ++i)
        {
            if (!seen[i])
            {
                throw std::runtime_error("PSC parts validation: missing part " + std::to_string(i) + " of " + std::to_string(total));
            }


        }
    }

    std::string maxDateOrEmpty(const std::vector<PscPartLink>& links)
    {
        std::string best;

        for (const auto p: links)
        {
            if (p.date > best)
            {
                best = p.date;
            }
        }
        return best;
    }

    std::vector<PscPartLink> selectLatestSnapshotPartsOrThrow(std::vector<PscPartLink> links)
    {
        const auto latest = maxDateOrEmpty(links);
        if (latest.empty())
        {
            throw std::runtime_error("No PSC parts found on listing page");
        }

        auto parts = filterAndSortParts(std::move(links), latest);

        if (parts.empty())
        {
            throw std::runtime_error("Latest PSC date found but no parts matched (unexpected)");
        }

        validateContiguousPartsOrThrow(parts);
        return parts;
    }

    std::optional<std::string> extractPscSingleZipHrefFromHtml(const std::string& html)
    {
        // Matches either:
        //   href="persons-with-significant-control-snapshot-2025-12-29.zip"
        //   href="https://download.companieshouse.gov.uk/persons-with-significant-control-snapshot-2025-12-29.zip"
        static const std::regex re(
            R"delim(href="((?:https?://download\.companieshouse\.gov\.uk/)?persons-with-significant-control-snapshot-(\d{4}-\d{2}-\d{2})\.zip)")delim",
            std::regex::icase);

        std::smatch m;
        if (!std::regex_search(html, m, re))
        {
            return std::nullopt;
        }
        return normalizeDownloadUrl(m[1].str());
    }

    std::optional<std::string> extractPscLatestSnapshotDateFromHtml(const std::string& html)
    {
        static const std::regex re(
            R"delim(persons-with-significant-control-snapshot-(\d{4}-\d{2}-\d{2})\.zip)delim",
            std::regex::icase);

        std::smatch m;
        if (!std::regex_search(html, m, re))
        {
            return std::nullopt;
        }
        return m[1].str();
    }

    bool BulkIngestService::ingestMonthlyFile(pg_conn* conn,
                                              std::optional<std::chrono::year_month_day> dateOverride,
                                              SeedStats* statsOut,
                                              const std::string& targetTable,
                                              bool truncateBeforeLoad) const
    {
        if (!isSimpleSqlIdentifier(targetTable))
        {
            PLOGE << "BulkIngestService::ingestMonthlyFile invalid target table: " << targetTable;
            return false;
        }
        auto targetDate = dateOverride.value_or(currentMonthStart());
        auto url = generateUrl(targetDate);

        fs::path downloadDir = mConfig.bulkDownloadDir;
        if (!downloadDir.empty() && !fs::exists(downloadDir))
        {
            std::error_code ec;
            fs::create_directories(downloadDir, ec);
            if (ec)
            {
                PLOGE << "[BulkIngestService] Failed to create download directory "
                      << downloadDir.string() << ": " << ec.message();
                return false;
            }
        }

        // create a templfile
        std::string filename = url.substr(url.find_last_of('/') + 1);
        fs::path filePath = downloadDir / filename;

        FileCleanUpGuard guard{filePath};

        PLOGI << "[BulkService] Target URL " << url;
        PLOGI << "[BulkService] Downloading to: " << filePath.string();
        
        bool downloaded = mHttp.downloadFile(url, filePath.string(),{},
            [](size_t now, size_t total)
            {
                // Log every 10MB
                if (total > 0 &&  now %(1024 * 1024* 10) == 0)
                {
                    PLOGD << "[BulkService] Downloaded "
                          << (now / (1024 * 1024)) << "MB / " << (total / (1024 * 1024)) << "MB";
                }
            });

        if (!downloaded)
        {
            PLOGE << "[BulkIngestService] Download failed from " << url;
            if (fs::exists(filePath))
            {
                fs::remove(filePath);
            }

            return false;
        }

        PLOGI << "[BulkService] Download complete";

        // Ingest
        try
        {
            if (!conn || PQstatus(conn) != CONNECTION_OK)
            {
                PLOGE << "BulkIngestService::ingestMonthlyFile invalid PG connection";
                return false;
            }

            if (truncateBeforeLoad)
            {
                const std::string truncateSql = "TRUNCATE TABLE " + targetTable;
                PGresult* truncateResult = PQexec(conn, truncateSql.c_str());
                if (!truncateResult || PQresultStatus(truncateResult) != PGRES_COMMAND_OK)
                {
                    const std::string err = PQerrorMessage(conn);
                    if (truncateResult) PQclear(truncateResult);
                    PLOGE << "[BulkIngestService] Truncate failed: " << err;
                    return false;
                }
                PQclear(truncateResult);
            }

            ZipSeedOptions options;
            options.allowPartialRow = true;
            options.targetTable = targetTable;
            bool result = seedCompaniesFromZipCsvFile(filePath.string(), conn, options, statsOut);

            if (result)
            {
                PLOGI << "[BulkService] Ingest successful";
            }
            else
            {
                PLOGE << "[BulkIngestService] Ingest failed";
            }

            return result;
        }
        catch (const std::exception& e)
        {
            PLOGE << "[BulkIngestService] Exception: " << e.what();
            if (fs::exists(filePath))
            {
                fs::remove(filePath);
            }
            return false;
        }

    }

                                        "  company_number,"
                                        "  name,"
                                        "  status,"
                                        "  sic_codes,"
                                        "  incorporation_date,"
                                        "  region,"

    // runs parser->COPY given a function that feeds decompressed bytes
    template<typename FeedFn>
    bool runStreamToCopy(FeedFn&& feedFn, pg_conn* conn, SeedStats* statsOut)
    {
        SeedStats stats{};
        pg::CopySession copy(conn, kCompaniesCsv);

        gz::Inflater inflater(gz::HeaderMode::GzipOnly);
        LineAccumulator acc;

        auto onDecompressed = [&](const uint8_t* p, size_t n)
        {
            std::string_view chunk(reinterpret_cast<const char*>(p), n);
            stats.linesTotal += acc.feed(chunk, [&](std::string_view line)
            {
                CompanyRecord rec;
                if (!parseCompanyNdjsonLine(line,rec))
                {
                    stats.linesBad++;
                    return;
                }
                const std::string csv = toCompanyCsvLine(rec);
                copy.putCsvLine(csv);
                stats.linesOk++;
            });
        };

        if (!feedFn(inflater, onDecompressed))
        {
            throw std::runtime_error("bulk stream failed");
        }

        inflater.finish(onDecompressed);
        acc.flush([&](std::string_view line)
        {
            stats.linesTotal++;
            CompanyRecord rec;
            if (!parseCompanyNdjsonLine(line,rec))
            {
                stats.linesBad++;
                return;
            }
            copy.putCsvLine(toCompanyCsvLine(rec));
            stats.linesOk++;
        });

        copy.finish();
        if (statsOut)
        {
            *statsOut = stats;
        }
        return true;
    }

    bool seedCompaniesFromNdjsonGzFile(const std::string& gzPath, pg_conn* conn, SeedStats* statsOut)
    {
        std::ifstream ifs(gzPath, std::ios::binary);
        if (!ifs)
        {
            return false;
        }

        auto feedFn = [&](gz::Inflater& inflater, const std::function<void(const uint8_t*, size_t)>& onOut) -> bool
        {
            std::vector<uint8_t> buffer(256 * 1024);
            while (ifs)
            {
                ifs.read(reinterpret_cast<char*>(buffer.data()), static_cast<std::streamsize>(buffer.size()));
                std::streamsize got = ifs.gcount();
                if (got > 0)
                {
                    inflater.feed(buffer.data(), static_cast<size_t>(got), onOut);
                }
            }
            return ifs.eof();
        };

        return runStreamToCopy(feedFn, conn, statsOut);
    }

    bool seedCompaniesFromNdjsonGzUrl(const IngestConfig& config, Http& http, const std::string& url, pg_conn* conn, SeedStats* statsOut)
    {
        auto feedFn = [&](gz::Inflater& inflater, const std::function<void(const uint8_t*, size_t)>& onOut) -> bool
        {
            bool ok = http.streamDownload(url, {},
                [&](const std::string& compressedChunk)
                {
                    if (!compressedChunk.empty())
                    {
                        inflater.feed(reinterpret_cast<const uint8_t*>(compressedChunk.data()), compressedChunk.size(), onOut);
                    }
                    return true;
                });
            return ok;
        };
        return runStreamToCopy(feedFn, conn, statsOut);
    }

#include "Deflater.h"

#include <cstring>
#include <stdexcept>

gz::Deflater::Deflater(int level, bool gzipWrapper, size_t outChunk) :
mLevel(level), mGzip(gzipWrapper), mOutBuffer(outChunk)
{
    std::memset(&mZStream, 0, sizeof(mZStream));
    // windowBits: +16 for gzip, +0 for zlib
    int windowBits = mGzip ? ( 16 + MAX_WBITS) : MAX_WBITS;
    int rc = deflateInit2(&mZStream, mLevel, Z_DEFLATED, windowBits, 8, Z_DEFAULT_STRATEGY);
    if (rc != Z_OK)
    {
        throwZlibError("deflateInit2", rc);
    }
    mInit = true;
}

gz::Deflater::~Deflater()
{
    if (mInit)
    {
        endStream();
        mInit = false;
    }
}

void gz::Deflater::ensureInit()
{
    if (!mInit)
    {
        int windowBits = mGzip ? ( 16 + MAX_WBITS ) : MAX_WBITS;
        int rc = deflateInit2(&mZStream, mLevel, Z_DEFLATED, windowBits, 8, Z_DEFAULT_STRATEGY);
        if (rc != Z_OK)
        {
            throwZlibError("deflateInit2", rc);
        }
        mInit = true;
        mFinished = false;;
    }
}

void gz::Deflater::endStream()
{
    if (mInit)
    {
        deflateEnd(&mZStream);
        mInit = false;
    }
}

void gz::Deflater::throwZlibError(const char* where, int code) const {
    const char* msg = mZStream.msg ? mZStream.msg : "";
    throw std::runtime_error(std::string(where) + " failed: " + std::to_string(code) + (msg[0] ? std::string(" (") + msg + ")" : ""));
}

size_t gz::Deflater::feed(const uint8_t* data, size_t len, const std::function<void(const uint8_t*, size_t)>& onOut)
{
    if (mFinished)
    {
        throw std::runtime_error("Deflater called after finished; call reset instead");
    }
    ensureInit();
    mZStream.next_in = const_cast<Bytef*>(reinterpret_cast<const Bytef*>(data));
    mZStream.avail_in = static_cast<uInt>(len);

    size_t produced = 0;
    while (mZStream.avail_in > 0)
    {
        mZStream.next_out = mOutBuffer.data();
        mZStream.avail_out = static_cast<uInt>(mOutBuffer.size());
        int rc = deflate(&mZStream, Z_NO_FLUSH);
        if (rc != Z_OK)
        {
            throwZlibError("deflate", rc);
        }
        size_t have = mOutBuffer.size() - mZStream.avail_out;
        if (have)
        {
            onOut(mOutBuffer.data(), have);
            produced += have;
        }
    }
    return produced;
}

size_t gz::Deflater::finish(const std::function<void(const uint8_t*, size_t)>& onOut)
{
    ensureInit();
    size_t produced = 0;
    int rc = Z_OK;
    do
    {
        mZStream.next_out = mOutBuffer.data();
        mZStream.avail_out = static_cast<uInt>(mOutBuffer.size());
        rc = deflate(&mZStream, Z_FINISH);
        if (rc == Z_STREAM_END || rc == Z_OK)
        {
            size_t have = mOutBuffer.size() - mZStream.avail_out;
            if (have)
            {
                onOut(mOutBuffer.data(), have);
                produced += have;
            }
        }
        else if (rc == Z_BUF_ERROR)
        {
            break;
        }
        else
        {
            throwZlibError("Deflate(Z_FINISH)", rc);
        }
    }
    while (rc != Z_STREAM_END);
    mFinished = true;
    return produced;
}

void gz::Deflater::reset(int level, bool gZipWrapper)
{
    endStream();
    mLevel = level;
    mGzip = gZipWrapper;
    std::memset(&mZStream, 0, sizeof(mZStream));
    int windowBits = mGzip ? ( 16 + MAX_WBITS ) : MAX_WBITS;
    int rc = deflateInit2(&mZStream, mLevel, Z_DEFLATED, windowBits, 8, Z_DEFAULT_STRATEGY);
    if (rc != Z_OK)
    {
        throwZlibError("deflateInit2(reset)", rc);
    }
    mInit = true;
    mFinished = false;
}

        : mJobs(std::move(jobs))
    {
    }

    SchedulerRunner::~SchedulerRunner()
    {
        stop();

    void SchedulerRunner::addJob(std::shared_ptr<ScheduledJob> job)
    {
        if (!job)
        {
            throw std::runtime_error("SchedulerRunner::addJob called with null job");
        }
        std::lock_guard<std::mutex> lock(mMutex);
        if (mRunning.load())
        {
            throw std::runtime_error("Cannot add jobs while SchedulerRunner is running");
        }
        mJobs.push_back(std::move(job));
    }

    void SchedulerRunner::start()
    {
        std::lock_guard<std::mutex> lock(mMutex);
        if (mRunning.load())
        {
            return;
        }
        if (mJobs.empty())
        {
            throw std::runtime_error("SchedulerRunner has no jobs to run");
        }

        mStopRequested.store(false);
        mRunning.store(true);
        mThreads.clear();
        mThreads.reserve(mJobs.size());
        for (const auto& job : mJobs)
        {
            mThreads.emplace_back([this, job]()
            {
                workerLoop(job);
            });
        }
    }

    void SchedulerRunner::stop()
    {
        if (!mRunning.load())
        {
            return;
        }

        mStopRequested.store(true);
        mCv.notify_all();
        for (auto& t : mThreads)
        {
            if (t.joinable())
            {
                t.join();
            }
        }
        mThreads.clear();
        mRunning.store(false);
    }

    bool SchedulerRunner::isRunning() const
    {
        return mRunning.load();
    }

    void SchedulerRunner::workerLoop(const std::shared_ptr<ScheduledJob>& job)
    {
        auto nextRun = std::chrono::steady_clock::now();

        while (!mStopRequested.load())
        {
            const auto now = std::chrono::steady_clock::now();
            if (now >= nextRun)
            {
                JobRunResult result = job->runOnce();
                if (!result.success)
                {
                    PLOGE << "[SchedulerRunner] job " << job->name()
                          << " failed: " << result.message;
                }
                nextRun = std::chrono::steady_clock::now() + std::max(job->interval(), std::chrono::milliseconds(1));
            }

            std::unique_lock<std::mutex> lock(mMutex);
            mCv.wait_until(lock, nextRun, [this]()
            {
                return mStopRequested.load();
            });
        }
    }
